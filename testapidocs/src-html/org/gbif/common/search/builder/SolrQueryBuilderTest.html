030  @Test
031  public void buildSearchRequestCombined() {
032    TestSearchRequest req = new TestSearchRequest();
033    req.addParameter(TestSearchParameter.HIGHERTAXON, "111");
034    req.addParameter(TestSearchParameter.RANK, Rank.SPECIES);
035    req.addParameter(TestSearchParameter.RANK, Rank.VARIETY);
036    req.addParameter(TestSearchParameter.EXTINCT, true);
037    req.addFacets(TestSearchParameter.HIGHERTAXON, TestSearchParameter.RANK);
038
039    SolrQuery query = buildQuery(req);
040
041    Set<String> facets = Sets.newHashSet(query.getFacetFields());
042    assertEquals(2, facets.size());
043    assertTrue(facets.contains("rank"));
044    assertTrue(facets.contains("higher_taxon_nub_key"));
045    assertFilterQueryContains(query, "{!tag=ffqextinct}extinct:true");
046    assertFilterQueryContains(query, "{!tag=ffqrank}(rank:26 OR rank:30)");
047    assertFilterQueryContains(query, "{!tag=ffqhigher_taxon_nub_key}higher_taxon_nub_key:111");
048  }

050  @Test

052    TestSearchRequest searchRequest = new TestSearchRequest();

061  @Test
062  public void buildSearchRequestFaceted() {
063    final UUID uuid1 = UUID.randomUUID();
064    final UUID uuid2 = UUID.randomUUID();
065
066    FacetedSearchRequest<TestSearchParameter> searchRequest = new FacetedSearchRequest<TestSearchParameter>();
067    searchRequest.setQ("puma");
068    searchRequest.addFacets(TestSearchParameter.values());
069    searchRequest.setMultiSelectFacets(true);
070    searchRequest.addParameter(TestSearchParameter.CHECKLIST, uuid1.toString());
071    searchRequest.addParameter(TestSearchParameter.CHECKLIST, uuid2.toString());
072
073    SolrQuery query = buildQuery(searchRequest);
074    assertNotNull(query.getFilterQueries());
075    assertNotNull(query.getFacetFields());
076    assertEquals(1, query.getFilterQueries().length);
077    final String x = query.getFilterQueries()[0];
078    assertTrue(x.contains("checklist_key:" + uuid1.toString()));
079    assertTrue(x.contains("checklist_key:" + uuid2.toString()));
080    // the search example doesnt have THREAT mapped as a facet, only regular param
081    assertEquals(TestSearchParameter.values().length - 1, query.getFacetFields().length);
082  }

154  private void assertFilterQueryContains(SolrQuery query, String filter) {
155    for (String fq : query.getFilterQueries()) {
156      if (filter.equals(fq)) {
157        return;
158      }
159    }
160    fail("Filter query does not contain " + filter + " : " + Joiner.on(" ").join(query.getFilterQueries()));
161  }

164  private SolrQuery buildQuery(FacetedSearchRequest<TestSearchParameter> searchRequest) {
165    SolrQueryBuilder<TestSearchResult, TestSearchParameter> requestBuilder =
166      SolrQueryBuilder.create(TestSearchParameter.class, TestSearchResult.class);
167    requestBuilder.withQueryBuilder(queryStringBuilder);
168    return requestBuilder.build(searchRequest);
169  }
170
171  private SolrQuery buildQuery(SearchRequest<TestSearchParameter> searchRequest) {
172    SolrQueryBuilder<TestSearchResult, TestSearchParameter> requestBuilder =
173      SolrQueryBuilder.create(TestSearchParameter.class, TestSearchResult.class);
174    requestBuilder.withQueryBuilder(queryStringBuilder);
175    return requestBuilder.build(searchRequest);
176  }
