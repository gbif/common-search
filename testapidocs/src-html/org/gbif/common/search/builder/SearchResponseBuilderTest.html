001package org.gbif.common.search.builder;
002
003import org.gbif.api.model.common.paging.PagingRequest;
004import org.gbif.api.model.common.search.SearchResponse;
005
006import java.util.List;
007
008import com.google.common.collect.Lists;
009import org.apache.solr.client.solrj.response.FacetField;
010import org.apache.solr.client.solrj.response.QueryResponse;
011import org.apache.solr.common.SolrDocumentList;
012import org.junit.Test;
013
014import static org.junit.Assert.assertEquals;
015import static org.mockito.Mockito.mock;
016import static org.mockito.Mockito.when;
017
018public class SearchResponseBuilderTest {
019
020  /**
021   * Tests that returned ordinal facet values are converted to actual enum names.
022   */
023  @Test
024  public void testBuild() throws Exception {
025    final SearchResponseBuilder builder =
026      SearchResponseBuilder.create(TestSearchResult.class, TestSearchResult.class, TestSearchParameter.class);
027
028    PagingRequest req = new PagingRequest(0,5);
029    QueryResponse queryResponse = mock(QueryResponse.class);
030    SolrDocumentList solrList = new SolrDocumentList();
031    when(queryResponse.getResults()).thenReturn(solrList);
032    List<FacetField> facets = Lists.newArrayList();
033    FacetField rankFacet = new FacetField("rank");
034    rankFacet.add("19", 87); // genus
035    rankFacet.add("14", 14); // family
036    rankFacet.add("1", 8);  // kingdom
037    facets.add(rankFacet);
038    when(queryResponse.getFacetFields()).thenReturn(facets);
039
040    SearchResponse<TestSearchResult, TestSearchParameter> resp = builder.build(req, queryResponse);
041
042    assertEquals(1, resp.getFacets().size());
043    assertEquals(TestSearchParameter.RANK, resp.getFacets().get(0).getField());
044    assertEquals(3, resp.getFacets().get(0).getCounts().size());
045    assertEquals("GENUS", resp.getFacets().get(0).getCounts().get(0).getName() );
046    assertEquals(87l, resp.getFacets().get(0).getCounts().get(0).getCount().longValue());
047    assertEquals("FAMILY", resp.getFacets().get(0).getCounts().get(1).getName() );
048    assertEquals(14l, resp.getFacets().get(0).getCounts().get(1).getCount().longValue());
049    assertEquals("KINGDOM", resp.getFacets().get(0).getCounts().get(2).getName() );
050    assertEquals(8l, resp.getFacets().get(0).getCounts().get(2).getCount().longValue());
051  }
052}
