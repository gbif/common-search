001package org.gbif.common.search.builder;
002
003import org.gbif.api.model.common.search.FacetedSearchRequest;
004import org.gbif.api.vocabulary.Rank;
005import org.gbif.common.search.util.SolrConstants;

007import java.util.regex.Matcher;
008
009import org.apache.solr.client.solrj.SolrQuery;
010import org.junit.Test;
011
012import static org.junit.Assert.assertEquals;
013import static org.junit.Assert.assertFalse;
014import static org.junit.Assert.assertNotNull;
015import static org.junit.Assert.assertTrue;
016
017/**
018 * Test class for common search operations.
019 */
020public class FullTextQueryStringBuilderTest {
021
022  private final FullTextQueryStringBuilder queryStringBuilder = FullTextQueryStringBuilder
023    .create(TestSearchResult.class);
024  private final String EXPECTED_PUMA_QUERY =
025    "(canonical_name:puma^100.0 OR class:puma^1.0 OR description_ft:puma^0.3 OR family:puma^2.0 OR genus:puma^2.0 OR kingdom:puma^1.0 OR order:puma^1.0 OR phylum:puma^1.0 OR scientific_name_ft:puma^10.0 OR scientific_name_ft:*puma*^0.2 OR species:puma^5.0 OR subgenus:puma^3.0 OR vernacular_name_ft:puma^6.0 OR vernacular_name_ft:*puma*^0.2)";
026
027  @Test
028  public void buildFullTextTest() {
029    String query = queryStringBuilder.build("puma");
030    assertTrue(query.contains("*"));
031  }
032
033
034  @Test
035  public void escapeSpecialChars() {
036    FacetedSearchRequest<TestSearchParameter> searchRequest = new FacetedSearchRequest<TestSearchParameter>();
037    searchRequest.addFacets(TestSearchParameter.values());
038    searchRequest.setMultiSelectFacets(true);
039    searchRequest.setQ("puma-con");
040    SolrQuery solrQuery = buildQuery(searchRequest);
041    assertEquals(EXPECTED_PUMA_QUERY.replaceAll("puma", "puma\\\\-con"), solrQuery.getQuery());
042  }
043
044  @Test
045  public void buildPhraseQueryFullTextTest() {
046    String query = queryStringBuilder.build("puma concolor");
047    assertFalse(query.contains("*"));
048  }
049
050  @Test
051  public void buildSearchRequestFaceted() {
052    FacetedSearchRequest<TestSearchParameter> searchRequest = new FacetedSearchRequest<TestSearchParameter>();
053    searchRequest.setQ("puma");
054    searchRequest.addFacets(TestSearchParameter.values());
055    searchRequest.setMultiSelectFacets(true);
056
057    SolrQuery solrQuery = buildQuery(searchRequest);
058    assertEquals(EXPECTED_PUMA_QUERY, solrQuery.getQuery());
059  }
060
061  @Test
062  public void buildSearchRequestFacetedNull() {
063    FacetedSearchRequest<TestSearchParameter> searchRequest = new FacetedSearchRequest<TestSearchParameter>();
064    searchRequest.setQ("puma");
065    searchRequest.addFacets(TestSearchParameter.CHECKLIST, TestSearchParameter.RANK);
066    searchRequest.setMultiSelectFacets(true);
067
068    SolrQuery solrQuery = buildQuery(searchRequest);
069    assertEquals(EXPECTED_PUMA_QUERY, solrQuery.getQuery());
070  }
071
072  @Test
073  public void buildTagFacetParamTest() {
074    String result = SolrConstants.FACET_FILTER_EX.replace(SolrConstants.TAG_FIELD_PARAM, "field");
075    String resultWrong =
076      SolrConstants.FACET_FILTER_EX.replace(SolrConstants.TAG_FIELD_PARAM, "field").replace("{", " ");
077    Matcher matcherOK = SolrConstants.FACET_FILTER_RGEX.matcher(result);
078    Matcher matcherError = SolrConstants.FACET_FILTER_RGEX.matcher(resultWrong);
079    Matcher matcherCleaner = SolrConstants.FACET_FILTER_RGEX_CLEAN.matcher(result);
080    assertTrue(matcherOK.matches());
081    assertFalse(matcherError.matches());
082    assertEquals("field", matcherCleaner.replaceAll(""));
083    assertNotNull(result);
084  }
085
086
087  @Test
088  public void builSearchRequestExample() {
089    TestSearchRequest searchRequest = new TestSearchRequest();
090    searchRequest.addRankFilter(Rank.SUBGENUS);
091
092    SolrQuery solrQuery = buildQuery(searchRequest);
093    assertEquals(SolrConstants.DEFAULT_QUERY, solrQuery.getQuery());
094  }
095
096  private SolrQuery buildQuery(FacetedSearchRequest<TestSearchParameter> searchRequest) {
097    SolrQueryBuilder<TestSearchResult, TestSearchParameter> requestBuilder =
098      SolrQueryBuilder.create(TestSearchParameter.class, TestSearchResult.class);
099    requestBuilder.withQueryBuilder(queryStringBuilder);
100    return requestBuilder.build(searchRequest);
101  }
