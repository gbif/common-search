001package org.gbif.common.search.solr.builders;
002
003import org.gbif.utils.file.FileUtils;
004import org.gbif.utils.file.ResourcesUtil;
005
006import java.io.File;
007import java.io.IOException;
008
009import com.google.common.base.Preconditions;
010import com.google.common.base.Strings;
011import com.google.common.base.Throwables;
012import org.apache.commons.lang3.ArrayUtils;
013import org.apache.solr.client.solrj.SolrClient;
014import org.apache.solr.client.solrj.embedded.EmbeddedSolrServer;
015import org.apache.solr.core.CoreContainer;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019/**
020 * Builds an instance of a {@link EmbeddedSolrServer}. The {@link SolrClient} can be created at a specific directory;
021 * set by the method withServerHomeDir, is the home directory is not specified a temporary directory is created.
022 */
023public class EmbeddedServerBuilder {
024
025
026  private static final Logger LOG = LoggerFactory.getLogger(EmbeddedServerBuilder.class);
027
028  // Default Solr configuration directory
029  private static final String SOLR_CONF = "solr/%s/conf/";
030
031  // Solr home system variable
032  private static final String SOLR_HOME = "solr.solr.home";
033
034  // Default Solr resources
035  private static final String[] SOLR_DEFAULT_RESOURCES = {"synonyms.txt", "protwords.txt", "stopwords.txt"};
036
037  // Default configuration resources
038  private static final String[] SOLR_CONF_DEFAULT_RESOURCES = ArrayUtils.addAll(new String[] {"schema.xml",
039    "solrconfig.xml"}, SOLR_DEFAULT_RESOURCES);
040
041  // Container name, defaults to collection1
042  private String coreName = "collection1";
043
044  // Flag to determine if the solr home directory must be deleted (used by tests cases mostly).
045  private boolean deleteOnExit;
046
047  // Solr home directory
048  private String serverHomeDir;
049
050  /**
051   * Builds a {@link EmbeddedSolrServer} instance. The server is created at the specified directory.
052   * If a solrHome directory is not specified a temporary directory is created (used by test cases).
053   */
054  public SolrClient build() {
055    if (Strings.isNullOrEmpty(serverHomeDir)) {
056      try {
057        serverHomeDir = FileUtils.createTempDir("solr-", "").getAbsolutePath();
058        LOG.debug("Using tmp solr server {}", serverHomeDir);
059      } catch (IOException e) {
060        Throwables.propagate(e);
061      }
062    }
063    // directory solr_test will be created below target in the maven build and populated with the needed files
064    File solrHome = new File(serverHomeDir);
065    if (deleteOnExit) {
066      solrHome.deleteOnExit();
067    }
068    copyResources(solrHome);
069    return builSolrServer(solrHome);
070  }
071
072
073  /**
074   * Solr coreName name, defaults to collection1.
075   * This is important since 4.4 as the solr directory has to have a subfolder with that name!
076   */
077  public EmbeddedServerBuilder withCoreName(String coreName) {
078    Preconditions.checkNotNull(coreName, "Invalid solr coreName name");
079    this.coreName = coreName;
080    return this;
081  }
082
083  /**
084   * Flag to determine if the solr home directory must be deleted (used by tests cases mostly).
085   */
086  public EmbeddedServerBuilder withDeleteOnExit(boolean deleteOnExit) {
087    this.deleteOnExit = deleteOnExit;
088    return this;
089  }
090
091  /**
092   * Solr home root directory. This is the parent folder of the actual core created.
093   */
094  public EmbeddedServerBuilder withServerHomeDir(String serverHomeDir) {
095    this.serverHomeDir = serverHomeDir;
096    return this;
097  }
098
099
100  /**
101   * Build the {@link SolrClient} instance, pointing to the solrHome directory.
102   */
103  private SolrClient builSolrServer(File solrHome) {
104    // create coreName
105    System.setProperty(SOLR_HOME, solrHome.getAbsolutePath());
106    CoreContainer cc = new CoreContainer(solrHome.getAbsolutePath());
107    cc.load();
108    SolrClient solrClient = new EmbeddedSolrServer(cc, coreName);
109    LOG.info("Created embedded solr server with solr dir {}", solrHome.getAbsolutePath());
110    return solrClient;
111  }
112
113  /**
114   * Copy default resources into the solrHome directory.
115   */
116  private void copyResources(File solrHome) {
117    try {
118      // since solr 4.4 the cores need to reside in a subfolder with that name
119      // see http://wiki.apache.org/solr/Solr.xml%204.4%20and%20beyond
120      File coreDir = new File(solrHome, coreName);
121      // copy solr resource files
122      ResourcesUtil.copy(solrHome, "solr/", false, "solr.xml");
123      // copy default configurations
124      File conf = new File(coreDir, "conf");
125      ResourcesUtil.copy(conf, "solr/default/", false, SOLR_DEFAULT_RESOURCES);
126      // copy specific configurations, overwriting above defaults
127      ResourcesUtil.copy(conf, String.format(SOLR_CONF,coreName), true, SOLR_CONF_DEFAULT_RESOURCES);
128      File coreProp = new File(coreDir, "core.properties");
129      if(!coreProp.exists()) {
130        // create empty core.properties in core dir
131        coreProp.createNewFile();
132      }
133    } catch (IOException e) {
134      throw new IllegalStateException(e);
135    }
136  }
137
138}
