016package org.gbif.common.search.model;
017
018import java.lang.annotation.ElementType;
019import java.lang.annotation.Retention;
020import java.lang.annotation.RetentionPolicy;
021import java.lang.annotation.Target;
022
023/**
024 * Annotates setter methods of fields that can be used as part of the full text search.
025 * This annotation should be consumed by SearchService implementations.
026 */
027@Target({ElementType.METHOD, ElementType.TYPE})
028@Retention(RetentionPolicy.RUNTIME)
029public @interface FullTextSearchField {
030
031  /**
032   * Name of the field used for exact matching of tokens, the absence of this field implies that the default field is
033   * used.
034   */
035  String exactMatchField() default "";
036
037  /**
038   * Value used for scoring exact matching.
039   */
040  double exactMatchScore() default 1.0d;
041
042  /**
043   * Default field name for full text search potentially doing a wildcard partial match.
044   */
045  String field();
046
047  /**
048   * Name of the highlighted field, is specified this field is used instead of the field attribute.
049   */
050  String highlightField() default "";
051
052  /**
053   * WildcardPadding pattern for the full text field.
054   */
055  WildcardPadding partialMatching() default WildcardPadding.NONE;
056
057  /**
058   * Value for scoring partial matching.
059   * This field is set regardless of the "partialMatching" values: BOTH, LEFT and RIGHT.
060   */
061  double partialMatchScore() default 0.5d;
062
063}
