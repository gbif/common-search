001package org.gbif.common.search.model;
002
003import java.util.List;
004
005/**
006 * A simple facade allowing a property to be a list of complex objects that can be treated as a simple String list
007 * for search highlighting. Return this interface in the getter of the annotated result object for the response builder
008 * to detect it.
009 */
010public interface HighlightableList {
011
012  /**
013   * Returns a list of string values in the same order as the original wrapped list the individual values are
014   * based on.
015   * @return a list of string values
016   */
017  List<String> getValues();
018
019  /**
020   * Replaces an existing value in the value list with a new value, e.g. containing markup for highlighting.
021   * @param index the position in the list of values
022   * @param newValue the new value to set
023   */
024  void replaceValue(int index, String newValue);
025}
