001package org.gbif.common.search.inject;
002
003import org.gbif.common.search.solr.SolrServerType;
004import org.gbif.common.search.solr.builders.CloudSolrServerBuilder;
005import org.gbif.common.search.solr.builders.EmbeddedServerBuilder;
006
007import java.net.MalformedURLException;
008
009import com.google.inject.Inject;
010import com.google.inject.PrivateModule;
011import com.google.inject.Provides;
012import com.google.inject.Scopes;
013import com.google.inject.Singleton;
014import org.apache.solr.client.solrj.SolrClient;
015import org.apache.solr.client.solrj.impl.HttpSolrClient;
016import org.apache.solr.client.solrj.impl.LBHttpSolrClient;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020
021/**
022 * Guice module providing a remote or embedded Solr server based on the modules configuration.
023 * Expects 5 properties to be bound:
024 * solr.server.type: must be set using literals available in {@link SolrServerType}, being SolrServerType.HTTP the
025 * default.
026 * solr.server: a http url to a remote Solr server, path for an embedded Solr or Zookeeper server url.
027 * solr.collection: Solr collection name (required by {@link CloudSolrServerBuilder})
028 * solr.server.http_lbservers: list of Solr http servers to create instance of {@link LBHttpSolrClient} (optional for
029 * {@link CloudSolrServerBuilder}).
030 * If empty an embedded server in /tmp will be created
031 * solr.delete: if null or true deletes any embedded server artifacts after java shuts down.
032 */
033public class SolrModule extends PrivateModule {
034
035  private static final Logger LOG = LoggerFactory.getLogger(SolrModule.class);
036
037  /**
038   * Provides a Solr client instance based on the SolrServerType.
039   */
040  @Provides
041  @Singleton
042  @Inject
043  public SolrClient providerSolr(SolrConfig cfg) {
044    if (cfg.serverType == SolrServerType.EMBEDDED) {
045      LOG.info("Using embedded solr server {}", cfg.serverHome);
046      EmbeddedServerBuilder serverBuilder = new EmbeddedServerBuilder()
047          .withServerHomeDir(cfg.serverHome).withDeleteOnExit(cfg.deleteOnExit).withCoreName(cfg.collection);
048      return serverBuilder.build();
049    } else if (cfg.serverType == SolrServerType.LBHTTP) {
050      try {
051        LOG.info("Using remote load-balanced solr server {}", cfg.serverHome);
052        return new LBHttpSolrClient(cfg.serverHome);
053      } catch (MalformedURLException e) {
054        throw new IllegalStateException(e);
055      }
056    } else if (cfg.serverType == SolrServerType.CLOUD) {
057      CloudSolrServerBuilder serverBuilder = new CloudSolrServerBuilder()
058          .withZkHost(cfg.serverHome);
059      return serverBuilder.build();
060    } else { // cfg.serverType == SolrServerType.HTTP)
061      LOG.info("Using remote solr server {}", cfg.serverHome);
062      return new HttpSolrClient(cfg.serverHome);
063    }
064  }
065
066  @Override
067  protected void configure() {
068    bind(SolrConfig.class).in(Scopes.SINGLETON);
069    expose(SolrClient.class);
070  }
071}
