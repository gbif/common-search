001package org.gbif.common.search.inject;
002
003import org.gbif.common.search.solr.SolrServerType;
004
005import com.google.inject.Inject;
006import com.google.inject.name.Named;
007
008/**
009 * Configurations for solr with optional injection that allow us to provide defaults for a temporary embedded server.
010 */
011public class SolrConfig {
012
013  @Inject
014  @Named("solr.server.type")
015  public SolrServerType serverType;
016
017  @Inject(optional = true)
018  @Named("solr.server")
019  public String serverHome = null;
020
021  @Inject(optional = true)
022  @Named("solr.collection")
023  public String collection = "collection1";
024
025  @Inject(optional = true)
026  @Named("solr.server.httpLbservers")
027  public String httpLBservers = null;
028
029
030  @Inject(optional = true)
031  @Named("solr.delete")
032  public boolean deleteOnExit = false;
033}
