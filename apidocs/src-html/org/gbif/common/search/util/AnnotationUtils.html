026/**
027 * Utility class for search annotations handling.
028 */
029public class AnnotationUtils {
030
031  private static final Logger LOG = LoggerFactory.getLogger(AnnotationUtils.class);
032
033  /**
034   * Private default/hidden constructor.
035   */
036  private AnnotationUtils() {
037    // empty block
038  }
039
040  /**
041   * Gets the first method annotated with the annotation parameter in the input clazz.
042   *
043   * @param clazz      to be analyzed
044   * @param annotation to be searched
045   */
046  public static Method getAnnotatedMethod(Class<?> clazz, Class<? extends Annotation> annotation) {
047    for (Method method : clazz.getMethods()) {
048      if (method.isAnnotationPresent(annotation)) {
049        return method;
050      }
051    }
052    return null;
053  }
054
055  /**
056   * Gets the name of the java property annotated with {@link Key}.
057   *
058   * @return name of the key field, null if no key field is defined.
059   */
060  public static String getKeyField(Class<?> annotatedClass) {
061    Method method = getAnnotatedMethod(annotatedClass, Key.class);
062    if (method != null) {
063      for (PropertyDescriptor descriptor : PropertyUtils.getPropertyDescriptors(annotatedClass)) {
064        Method writeMethod = descriptor.getWriteMethod();
065        // PropertyDescriptor is not used directly to get the annotated method because a bug BeanUtils class.
066        if (writeMethod != null && writeMethod.getName().equals(method.getName())) {
067          return descriptor.getName();
068        }
069      }
070    }
071    LOG.error("Key field not found in class {}", annotatedClass.getName());
072    return null;
073  }
074
075
076  /**
077   * Helper method for initializing the map of SolrFields <-> Facet definitions using the Field Solr annotations.
078   *
079   * @param annotatedClass that contains the Solr @Field annotation.
080   *
081   * @return a map with SolrFieldName <-> FieldFacet definition.
082   */
083  public static Map<String, FacetField> initFacetFieldDefs(Class<?> annotatedClass) {
084    Map<String, FacetField> fieldDefs = Maps.newHashMap();
085    if (annotatedClass.isAnnotationPresent(SearchMapping.class)) {
086      for (FacetField f : annotatedClass.getAnnotation(SearchMapping.class).facets()) {
087        fieldDefs.put(f.field(), f);
088      }
089    }
090    return fieldDefs;
091  }
092
093
094  /**
095   * Helper method for initializing the map of SolrFields <-> Java fields mapped using an annotation type.
096   * The facet names are uppercased to allow case insensitive faceted parameters.
097   * Facet fields are case sensitive.
098   */
099  public static <P extends Enum<?> & SearchParameter> BiMap<String, P> initFacetFieldsPropertiesMap(
100      Class<?> annotatedClass, Class<P> enumClass) {
101    BiMap<String, P> fieldsPropertiesMap = HashBiMap.create();
102
103    if (annotatedClass.isAnnotationPresent(SearchMapping.class)) {
104      for (FacetField annotation : annotatedClass.getAnnotation(SearchMapping.class).facets()) {
105        fieldsPropertiesMap.put(annotation.field(), VocabularyUtils.lookupEnum(annotation.name(), enumClass));
106      }
107    }
108    for (Method method : annotatedClass.getMethods()) {
109      if (method.isAnnotationPresent(FacetField.class)) {
110        FacetField annotation = method.getAnnotation(FacetField.class);
111        fieldsPropertiesMap.put(annotation.field(), VocabularyUtils.lookupEnum(annotation.name(), enumClass));
112      }
113    }
114    return fieldsPropertiesMap;
115  }
116
117  /**
118   * Helper method for initializing the map of SolrFields <-> Java fields mapped using the Field Solr annotation
119   * found on setter methods.
120   *
121   * @param annotatedClass that contains the Solr @Field annotation on setters
122   *
123   * @return a map with SolrFieldName <-> PropertyName.
124   */
125  public static BiMap<String, String> initFieldsPropertiesMap(Class<?> annotatedClass) {
126    BiMap<String, String> fieldsPropertiesMap = HashBiMap.create();
127
128    // we iterate over all methods and look for our custom annotations
129    Method[] methods = annotatedClass.getMethods();
130    for (Method m : methods) {
131      if (m.isAnnotationPresent(Field.class)) {
132        // they should be on setters only
133        Field annotation = m.getAnnotation(Field.class);
134        String solrFieldName = annotation.value();
135        // remove set and lowercase first char
136        String propertyName =
137            new StringBuilder().append(Character.toLowerCase(m.getName().substring(3).charAt(0)))
138            .append(m.getName().substring(4)).toString();
139        fieldsPropertiesMap.put(solrFieldName, propertyName);
140      }
141    }
142
143    return fieldsPropertiesMap;
144  }
145
146  /**
147   * Helper method for initializing the map of SolrFields <-> Java fields mapped using an annotation type.
148   */
149  public static List<FullTextSearchField> initFullTextFieldsPropertiesMap(Class<?> annotatedClass) {
150    List<FullTextSearchField> fullTextSearchFields = new ArrayList<FullTextSearchField>();
151    if (annotatedClass.isAnnotationPresent(SearchMapping.class)) {
152      fullTextSearchFields.addAll(Arrays.asList(annotatedClass.getAnnotation(SearchMapping.class).fulltextFields()));
153    }
154    for (PropertyDescriptor descriptor : PropertyUtils.getPropertyDescriptors(annotatedClass)) {
155      Method writeMethod = descriptor.getWriteMethod();
156      if (writeMethod != null && writeMethod.isAnnotationPresent(FullTextSearchField.class)) {
157        fullTextSearchFields.add(writeMethod.getAnnotation(FullTextSearchField.class));
158      }
159    }
160    return fullTextSearchFields;
161  }
