077  /**
078   * Builds SolrQuery suitable to the TermsComponent.
079   *
080   * @param fieldName solr field to be searched
081   * @param limit maximum number of results
082   * @return a {@link SolrQuery}
083   */
084  public static SolrQuery buildTermQuery(String fieldName, Integer limit) {
085    SolrQuery solrQuery = new SolrQuery();
086    solrQuery.setParam(CommonParams.QT, TERMS_COMP_PATH);
087    solrQuery.setParam(TermsParams.TERMS, true);
088    solrQuery.setParam(TermsParams.TERMS_LIMIT, limit.toString());
089    solrQuery.setParam(TermsParams.TERMS_FIELD, fieldName);
090    solrQuery.setParam(TermsParams.TERMS_SORT, TermsParams.TERMS_SORT_INDEX);
091    return solrQuery;
092  }

094  /**
095   * Builds SolrQuery suitable to the TermsComponent.
096   *
097   * @param prefix search term
098   * @param fieldName solr field to be searched
099   * @param limit maximum number of results
100   * @return a {@link SolrQuery}
101   */
102  public static SolrQuery buildTermQuery(String prefix, String fieldName, Integer limit) {
103    SolrQuery solrQuery = buildTermQuery(fieldName, limit);
104    solrQuery.setParam(TermsParams.TERMS_PREFIX_STR, prefix);
105    return solrQuery;
106  }

108  /**
109   * Removes all the consecutive whitespaces and leave only 1.
110   */
111  public static String clearConsecutiveBlanks(String query) {
112    return REGEX_MULTIPLE_BLANKS.matcher(query).replaceAll(SINGLE_BLANK);
113  }

115  /**
116   * If the query parameter is empty returns the default query "*".
117   *
118   * @return default query if q parameter is empty
119   */
120  public static String emptyToDefaultQuery(String q) {
121    // return default query for empty queries
122    String qValue = Strings.nullToEmpty(q).trim();
123    if (Strings.isNullOrEmpty(qValue)) {
124      return DEFAULT_FILTER_QUERY;
125    }
126    return qValue;
127  }

129  /**
130   * Utility method that return the exact matching field to use.
131   * If field.exactMatchField is null or empty, return field.field() else return field.exactMatchField().
132   */
133  public static String getExactMatchField(FullTextSearchField field) {
134    return Strings.isNullOrEmpty(field.exactMatchField()) ? field.field() : field.exactMatchField();
135  }

137  /**
138   * According to the FullTextSearchField.partialMatching() returns a search pattern that could contains wildcards.
139   *
140   * @param field the Annotation to be analyzed
141   * @param query the input search pattern
142   */
143  public static String getMatchPatternAndScore(FullTextSearchField field, String query) {
144    if (field.partialMatching() == WildcardPadding.BOTH) {
145      return DEFAULT_FILTER_QUERY + query + DEFAULT_FILTER_QUERY + SCORE_OP + field.partialMatchScore();
146    } else if (field.partialMatching() == WildcardPadding.LEFT) {
147      return DEFAULT_FILTER_QUERY + query + SCORE_OP + field.partialMatchScore();
148    } else if (field.partialMatching() == WildcardPadding.RIGHT) {
149      return query + DEFAULT_FILTER_QUERY + SCORE_OP + field.partialMatchScore();
150    } else { // NONE
151      return query;
152    }
153  }

170  /**
171   * Escapes a query value and transform it into a phrase query if necessary.
172   */
173  public static String parseQueryValue(final String q) {
174    // return default query for empty queries
175    String qValue = Strings.nullToEmpty(q).trim();
176    if (Strings.isNullOrEmpty(qValue)) {
177      return DEFAULT_FILTER_QUERY;
178    }
179    if (isRangeQuery(qValue)) {
180      String[] rangeValue = qValue.split(",");
181      if (rangeValue.length == 2) {
182        return String.format(SolrConstants.RANGE_FORMAT, rangeValue[0].trim(), rangeValue[1].trim());
183      }
184    } else {
185      // If default query was sent, must not be escaped
186      if (!qValue.equals(DEFAULT_FILTER_QUERY)) {
187        qValue = clearConsecutiveBlanks(qValue);
188        qValue = escapeQuery(qValue);
189      }
190      // make it a phrase query if it contains blanks
191      if (qValue.contains(BLANK)) {
192        qValue = toPhraseQuery(qValue);
193      }
194    }
195    return qValue;
196  }
197
198  /**
199   * Escape special characters and Solr reserved words.
200   */
201  public static String escapeQuery(String value) {
202    if (value.equals(QueryParser.Operator.AND.name()) || value.equals(QueryParser.Operator.OR.name())
203        || value.equals(SolrConstants.SOLR_NOT_OP)) {
204      return toPhraseQuery(ClientUtils.escapeQueryChars(value));
205    }
206    return ClientUtils.escapeQueryChars(value);
207  }
208
209  /**
210   * If the value parameter starts with {@link SearchConstants#NOT_OP} it is removed.
211   */
212  public static String removeNegation(String value) {
213    if (value.startsWith(NOT_OP)) {
214      return REGEX_NOT_OP.matcher(value).replaceFirst("");
215    }
216    return value;
217  }

220  /**
221   * Helper method that sets the parameters for a paginated query.
222   *
223   * @param pageable the Pageable used to extract the parameters
224   * @param solrQuery this object is modified adding the pagination parameters
225   */
226  public static void setQueryPaging(Pageable pageable, SolrQuery solrQuery) {
227    Long offset = pageable.getOffset();
228    solrQuery.setRows(Math.min(pageable.getLimit(), MAX_PAGE_SIZE));
229    solrQuery.setStart(Math.max(0, offset.intValue()));
230  }

237   * @param maxPageSize the maximum page size allowed

267  /**
268   * Sets the sort order query information.
269   *
270   * @param solrQuery to be modified.
271   */
272  public static void setSortOrder(SolrQuery solrQuery, Map<String, SolrQuery.ORDER> sortOrder) {
273    if (sortOrder != null) {
274      for (Map.Entry<String, SolrQuery.ORDER> so : sortOrder.entrySet()) {
275        solrQuery.addSort(so.getKey(), so.getValue());
276      }
277    }
278  }

280  /**
281   * Produces a query with the form: "(query)^boostValue".
282   *
283   * @param query to be boosted
284   * @param boostValue Solr boost factor
285   * @return a query in format "(query)^boostValue."
286   */
287  public static String toBoostedQuery(String query, Integer boostValue, boolean withParentheses) {
288    if (withParentheses) {
289      return toParenthesesQuery(query + SCORE_OP + boostValue);
290    }
291    return query + SCORE_OP + boostValue;
292  }

297  public static String toDateQueryFormat(Date value) {

302  /**
303   * Transforms the query into query surrounded by parentheses.
304   */
305  public static String toParenthesesQuery(String query) {
306    return String.format(PARENTHESES_EXP_FORMAT, query);
307  }

309  /**
310   * Adds the apostrophes to convert the input pattern into a phrase query pattern.
311   *
312   * @param query the input search pattern
313   */
314  public static String toPhraseQuery(String query) {
315    return APOSTROPHE + query + APOSTROPHE;
316  }

318  /**
319   * Sets the Solr facet.method for the field parameter according to the method parameter.
320   */
321  public static void setFacetMethod(SolrQuery solrQuery, String field, FacetField.Method facetFieldMethod) {
322    solrQuery.setParam(String.format(FACET_METHOD_FMT, field), FACET_METHOD_MAP.get(facetFieldMethod));
323  }
