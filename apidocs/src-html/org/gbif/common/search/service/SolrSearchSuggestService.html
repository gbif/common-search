013package org.gbif.common.search.service;
014
015
016import org.gbif.api.model.common.search.FacetedSearchRequest;
017import org.gbif.api.model.common.search.SearchParameter;
018import org.gbif.api.model.common.search.SearchRequest;
019import org.gbif.api.service.common.SuggestService;
020import org.gbif.common.search.builder.SearchResponseBuilder;
021import org.gbif.common.search.builder.SolrQueryBuilder;
022import org.gbif.common.search.builder.SuggestQueryStringBuilder;
023import org.gbif.common.search.exception.SearchException;
024
025import java.io.IOException;
026import java.util.List;
027import java.util.Map;
028
029import javax.annotation.Nullable;
030
031import org.apache.solr.client.solrj.SolrClient;
032import org.apache.solr.client.solrj.SolrQuery;
033import org.apache.solr.client.solrj.SolrServerException;
034import org.apache.solr.client.solrj.response.QueryResponse;
035import org.slf4j.Logger;
036import org.slf4j.LoggerFactory;
037
038
039/**
040 * Extension of {@link SolrSearchService} that implements the {@link org.gbif.api.service.common.SuggestService}.
041 * This class is generic implementation that can be re-used for different and more implementations.
042 *
043 * @param <T> Type of request object
044 * @param <R> Type of response object
045 * @param <P> the search parameter enum
046 */
047public class SolrSearchSuggestService<T, P extends Enum<?> & SearchParameter, ST extends T, R extends FacetedSearchRequest<P>, SUGT, SSUGT extends SUGT, RSUG extends SearchRequest<P>>
048  extends SolrSearchService<T, P, ST, R> implements SuggestService<SUGT, P, RSUG> {
049
050  // Logger
051  private static final Logger LOG = LoggerFactory.getLogger(SolrSearchSuggestService.class);
052
053  // Cached instance of query builder.
054  private final SolrQueryBuilder<ST, P> suggestQueryBuilder;
055
056  private static final int DEFAULT_SUGGEST_LIMIT = 10;
057
058  private final Class<SSUGT> suggestAnnotatedClass;
059
060  /**
061   * Default constructor.
062   *
063   * @param solrClient Solr server instance, this abstract type is used because it can hold instance of
064   *        HttpSolrClient or EmbeddedSolrServer
065   * @param searchType of the results content
066   * @param primarySortOrder ordered fields used for an optional sort order in every search
067   */
068  public SolrSearchSuggestService(SolrClient solrClient, Class<T> searchType,
069    Class<ST> searchSolrType, Class<P> searchParameterType, Map<String, SolrQuery.ORDER> primarySortOrder,
070    Class<SSUGT> suggestAnnotatedClass) {
071    super(solrClient, searchType, searchSolrType, searchParameterType, primarySortOrder);
072    this.suggestAnnotatedClass = suggestAnnotatedClass;
073    suggestQueryBuilder =
074      SolrQueryBuilder.create(searchParameterType, searchSolrType).withQueryBuilder(
075        SuggestQueryStringBuilder.create(suggestAnnotatedClass));
076  }
077
078
079  /**
080   * Constructor for regular search/suggest operations.
081   * Doesn't contain the default order for suggest results.
082   *
083   * @param solrServer Solr server instance, this abstract type is used because it can hold instance of
084   *        CommonsHttpSolrServer or EmbeddedSolrServer
085   * @param requestHandler specific Solr request handler to be used
086   * @param searchType of the results content
087   * @param primarySortOrder ordered fields used for an optional sort order in every search
088   */
089  public SolrSearchSuggestService(SolrClient solrClient, @Nullable final String requestHandler, Class<T> searchType,
090    Class<ST> searchSolrType, Class<P> searchParameterType, Map<String, SolrQuery.ORDER> primarySortOrder,
091    Class<SSUGT> suggestAnnotatedClass) {
092    super(solrClient, requestHandler, searchType, searchSolrType, searchParameterType, primarySortOrder);
093    this.suggestAnnotatedClass = suggestAnnotatedClass;
094    suggestQueryBuilder =
095      SolrQueryBuilder.create(searchParameterType, searchSolrType).withRequestHandler(requestHandler)
096        .withQueryBuilder(SuggestQueryStringBuilder.create(suggestAnnotatedClass));
097  }
098
099
100  /**
101   * Full constructor.
102   *
103   * @param solrClient Solr client instance, this abstract type is used because it can hold instance of
104   *        CommonsHttpSolrServer or EmbeddedSolrServer
105   * @param requestHandler specific Solr request handler to be used
106   * @param searchType of the results content
107   * @param primarySortOrder ordered fields used for an optional sort order in every search
108   * @param suggestSortOrder ordered fields used for an optional sort order in every suggest operation
109   */
110  public SolrSearchSuggestService(SolrClient solrClient, @Nullable final String requestHandler, Class<T> searchType,
111    Class<ST> searchSolrType, Class<P> searchParameterType, Map<String, SolrQuery.ORDER> primarySortOrder,
112    Map<String, SolrQuery.ORDER> suggestSortOrder, Class<SSUGT> suggestAnnotatedClass) {
113    super(solrClient, requestHandler, searchType, searchSolrType, searchParameterType, primarySortOrder);
114    this.suggestAnnotatedClass = suggestAnnotatedClass;
115    suggestQueryBuilder =
116      SolrQueryBuilder.create(searchParameterType, searchSolrType).withRequestHandler(requestHandler)
117        .withQueryBuilder(SuggestQueryStringBuilder.create(suggestAnnotatedClass))
118        .withPrimarySortOrder(suggestSortOrder);
119  }
120
121  @Override
122  public List<SUGT> suggest(RSUG suggestRequest) {
123    if (suggestRequest.getLimit() < 1) {
124      LOG.debug("Suggest request with limit {} found. Reset to default {}", suggestRequest.getLimit(),
125        DEFAULT_SUGGEST_LIMIT);
126      suggestRequest.setLimit(DEFAULT_SUGGEST_LIMIT);
127    }
128    if (suggestRequest.getOffset() > 0) {
129      LOG.debug("Suggest request with offset {} found", suggestRequest.getOffset());
130    }
131
132    try {
133      SolrQuery solrQuery = suggestQueryBuilder.build(suggestRequest);
134      QueryResponse queryResponse = getSolrClient().query(solrQuery);
135      return (List<SUGT>) SearchResponseBuilder.buildSuggestReponse(suggestRequest, queryResponse,
136        suggestAnnotatedClass)
137        .getResults();
138
139    } catch (SolrServerException | IOException e) {
140      LOG.error("Error executing/building the request", e);
141      throw new SearchException(e);
142    }
143  }
