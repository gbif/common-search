013package org.gbif.common.search.service;
014
015import org.gbif.api.model.common.search.FacetedSearchRequest;
016import org.gbif.api.model.common.search.SearchParameter;
017import org.gbif.api.model.common.search.SearchResponse;
018import org.gbif.api.service.common.SearchService;
019import org.gbif.common.search.builder.FullTextQueryStringBuilder;
020import org.gbif.common.search.builder.SearchResponseBuilder;
021import org.gbif.common.search.builder.SolrQueryBuilder;
022import org.gbif.common.search.exception.SearchException;
023
024import java.io.IOException;
025import java.util.LinkedHashMap;
026import java.util.Map;
027
028import javax.annotation.Nullable;
029
030import org.apache.solr.client.solrj.SolrClient;
031import org.apache.solr.client.solrj.SolrQuery;
032import org.apache.solr.client.solrj.SolrServerException;
033import org.apache.solr.client.solrj.response.QueryResponse;
034import org.slf4j.Logger;
035import org.slf4j.LoggerFactory;
036
037
038/**
039 * Generic {@link SearchService} that encapsulates the communication with a {@link SolrClient}.
040 * The basic implementation of the service is a search operation that holds all the valid parameters.
041 * The response returned is basically information taken from the Solr {@link QueryResponse}.
042 * During the initialization time the class expects a {@link SolrClient} instance which
043 * depending on the configuration parameters is an remote or a embedded Solr server.
044 * Additionally, this class internally implements the necessary logic for accessing the mapped fields between Solr and
045 * the request object values; most of this is implemented by using annotations, enum types and java introspection.
046 * All the mapping handling is done by {@link SearchResponseBuilder} and
047 * {@link org.gbif.common.search.builder.SolrQueryBuilder}.
048 *
049 * @param <T> the type of the results that the search operation return using a {@link SearchResponse}<T>
050 * @param <P> the search parameter enum also used for facets
051 */
052public class SolrSearchService<T, P extends Enum<?> & SearchParameter, ST extends T, R extends FacetedSearchRequest<P>>
053    implements SearchService<T, P, R> {
054
055  private static final Logger LOG = LoggerFactory.getLogger(SolrSearchService.class);
056
057  /*
058   * Solr server instance, this abstract type is used because it can hold instance of:
059   * CommonsHttpSolrServer or EmbeddedSolrServer.
060   */
061  private final SolrClient solrClient;
062
063  // Cached instance of response builder.
064  // This instance is used for cloning subsequent usages of this object without the penalty recreate similar instances.
065  private final SearchResponseBuilder<T, ST, P> responseBuilder;
066
067  private final SolrQueryBuilder<ST, P> searchQueryBuilder;
068
069  /**
070   * Default constructor.
071   *
072   * @param solrClient Solr server instance, this abstract type is used because it can hold instance of
073   *        CommonsHttpSolrServer or EmbeddedSolrServer
074   * @param type of the results content
075   * @param solrType Solr response type
076   * @param enumSearchParamType type of the enum search parameter
077   * @param primarySortOrder ordered fields used for an optional sort order in every search
078   */
079  public SolrSearchService(SolrClient solrClient, Class<T> type, Class<ST> solrType,
080    Class<P> enumSearchParamType, Map<String, SolrQuery.ORDER> primarySortOrder) {
081    this.solrClient = solrClient;
082    FullTextQueryStringBuilder fullTextQueryBuilder = FullTextQueryStringBuilder.create(solrType);
083    responseBuilder = SearchResponseBuilder.create(type, solrType, enumSearchParamType);
084    responseBuilder.withHighlightFields(fullTextQueryBuilder.getHighlightedFields());
085    searchQueryBuilder = SolrQueryBuilder.create(enumSearchParamType, solrType);
086    searchQueryBuilder.withQueryBuilder(fullTextQueryBuilder);
087    searchQueryBuilder.withPrimarySortOrder(new LinkedHashMap<String, SolrQuery.ORDER>(primarySortOrder));
088  }
089
090  public SolrSearchService(SolrClient solrClient, Class<T> type, Class<ST> solrType,
091                           Class<P> enumSearchParamType) {
092    this.solrClient = solrClient;
093    FullTextQueryStringBuilder fullTextQueryBuilder = FullTextQueryStringBuilder.create(solrType);
094    responseBuilder = SearchResponseBuilder.create(type, solrType, enumSearchParamType);
095    responseBuilder.withHighlightFields(fullTextQueryBuilder.getHighlightedFields());
096    searchQueryBuilder = SolrQueryBuilder.create(enumSearchParamType, solrType);
097    searchQueryBuilder.withQueryBuilder(fullTextQueryBuilder);
098  }
099
100  /**
101   * Full constructor.
102   *
103   * @param solrClient Solr client instance, this abstract type is used because it can hold instance of
104   *        CommonsHttpSolrServer or EmbeddedSolrServer
105   * @param requestHandler specific Solr request handler to be used
106   * @param type of the results content
107   * @param primarySortOrder ordered fields used for an optional sort order in every search
108   */
109  public SolrSearchService(SolrClient solrClient, @Nullable final String requestHandler, Class<T> type,
110    Class<ST> solrType,
111    Class<P> enumSearchParamType, Map<String, SolrQuery.ORDER> primarySortOrder) {
112    this(solrClient, type, solrType, enumSearchParamType, primarySortOrder);
113    searchQueryBuilder.withRequestHandler(requestHandler);
114  }
115
116
117  /**
118   * Issues a SolrQuery and converts the response to a SearchResponse object. Besides, the facets and paging
119   * parameter and responses are handled in the request and response objects.
120   *
121   * @param searchRequest the searchRequest that contains the search parameters
122   * @return the SearchResponse of the search operation
123   */
124  @Override
125  public SearchResponse<T, P> search(final R searchRequest) {
126
127    try {
128      // Defensive copy: done because the build method is not thread safe.
129      SolrQueryBuilder<ST, P> requestBuilder = searchQueryBuilder.getCopy();
130      SolrQuery solrQuery = requestBuilder.build(searchRequest);
131
132      // Executes the search operation in Solr
133      LOG.debug("Solr query executed: {}", solrQuery);
134      final QueryResponse queryResponse = solrClient.query(solrQuery);
135
136      // Defensive copy: done because the build method is not thread safe.
137      return responseBuilder.getCopy().build(searchRequest, queryResponse);
138    } catch (SolrServerException e) {
139      if (e.getRootCause() instanceof IllegalArgumentException) {
140        LOG.error("Bad search request", e);
141        throw (IllegalArgumentException) e.getRootCause();
142      } else {
143        LOG.error("Error executing the search operation", e);
144        throw new SearchException(e);
145      }
146    } catch (IOException e) {
147      LOG.error("Error executing the search operation", e);
148      throw new SearchException(e);
149    }
150  }
151
152  /**
153   * @return the responseBuilderBase
154   */
155  protected SearchResponseBuilder<T, ST, P> getResponseBuilder() {
156    return responseBuilder;
157  }
158
159
160  /**
161   * @return the solrClient
162   */
163  protected SolrClient getSolrClient() {
164    return solrClient;
165  }
166}
