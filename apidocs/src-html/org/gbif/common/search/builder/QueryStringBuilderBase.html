035/**
036 * Builder class that helps in the creation process of query patterns for classes annotated with
037 * {@link org.gbif.common.search.model.FullTextSearchField}.
038 */
039public abstract class QueryStringBuilderBase {
040
041  protected static final Logger LOG = LoggerFactory.getLogger(QueryStringBuilderBase.class);
042
043  // Solr parameter place holder
044  protected static final String QUERY_PLACE_HOLDER = "$q";
045
046  // query template patterns to build queries by replacing the above placeholder
047  protected String queryTemplate = QUERY_PLACE_HOLDER;
048  protected String phraseQueryTemplate = QUERY_PLACE_HOLDER;
049
050  protected final List<String> highlightedFields = Lists.newArrayList();
051
052  protected QueryStringBuilderBase(Class<?> annotatedClass) {
053    initTemplates(annotatedClass);
054    LOG.info("Query patterns generated for simple / phrase searches : {} / {}", queryTemplate, phraseQueryTemplate);
055  }
056
057  /**
058   * Creates a string containing the full text query expression.
059   *
060   * @param searchRequest used to extract parameters information.
061   */
062  public String build(final SearchRequest<?> searchRequest) {
063    return build(searchRequest.getQ());
064  }
065
066  /**
067   * Creates a string containing the full text query expression.
068   */
069  public String build(String q) {
070    String parsedQ = parseQueryValue(q);
071    String generatedQuery = getSearchPattern(parsedQ).replace(QUERY_PLACE_HOLDER, parsedQ);
072    LOG.debug("Solr query generated for fulltext search: {}", generatedQuery);
073    return generatedQuery;
074  }
075
076  /**
077   * Gets list of the fields for highlighting.
078   */
079  public List<String> getHighlightedFields() {
080    return highlightedFields;
081  }
082
083  /**
084   * Builds a string literal surround by parenthesis (or empty) of the query pattern.
085   *
086   * @param queryComponets list of query components in the form field:pattern.
087   *
088   * @return a string with all the query components.
089   */
090  protected static String buildQueryExpression(List<String> queryComponets) {
091    if (!queryComponets.isEmpty()) {
092      return '(' + PARAMS_OR_JOINER.join(queryComponets) + ')';
093    }
094    return "";
095  }
096
097  /**
098   * Helper method to determine if the query pattern to be used is the full text or the phrase query pattern.
099   */
100  protected String getSearchPattern(String qValue) {
101    if (qValue.contains(BLANK)) { // is a phrase query
102      return phraseQueryTemplate;
103    } else if (qValue.equals(DEFAULT_FILTER_QUERY)) {
104      return DEFAULT_QUERY;
105    }
106    return queryTemplate;
107  }
108
109  /**
110   * Helper method that sets the parameters for full text search param "q".
111   * It will also add the optional query function.
112   */
113  protected abstract void initTemplates(Class<?> annotatedClass);
114
115}
