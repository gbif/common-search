013package org.gbif.common.search.builder;
014
015import org.gbif.common.search.model.SuggestMapping;
016import org.gbif.common.search.util.QueryUtils;
017import org.gbif.common.search.util.SolrConstants;
018
019import java.util.ArrayList;
020import java.util.Collection;
021import org.apache.solr.client.solrj.util.ClientUtils;
022
023import static org.gbif.common.search.util.QueryUtils.PARAMS_JOINER;
024import static org.gbif.common.search.util.QueryUtils.PARAMS_OR_JOINER;
025import static org.gbif.common.search.util.QueryUtils.clearConsecutiveBlanks;
026import static org.gbif.common.search.util.QueryUtils.toBoostedQuery;
027import static org.gbif.common.search.util.QueryUtils.toParenthesesQuery;
028import static org.gbif.common.search.util.QueryUtils.toPhraseQuery;
029
030/**
031 * Builder class that helps in the creation process of query patterns for classes annotated with
032 * {@link org.gbif.common.search.model.SuggestMapping}.
033 */
034public class SuggestQueryStringBuilder extends QueryStringBuilderBase {
035
036  private static final Integer PHRASE_QUERY_BOOST = 1000;
037
038  private static final Integer PARTIAL_QUERY_BOOST = 300;
039
040  private static final Integer PARTIAL_BOOST_DECREMENT = 100;
041
042  private String startPhraseQueryTemplate;
043
044  /**
045   * Private default constructor.
046   */
047  private SuggestQueryStringBuilder(Class<?> annotatedClass) {
048    super(annotatedClass);
049  }
050
051  public static SuggestQueryStringBuilder create(Class<?> annotatedClass) {
052    return new SuggestQueryStringBuilder(annotatedClass);
053  }
054
055
056  /*
057   * (non-Javadoc)
058   * @see org.gbif.common.search.builder.QueryStringBuilderBase#build(java.lang.String)
059   */
060  @Override
061  public String build(String q) {
062    final String qValue = QueryUtils.emptyToDefaultQuery(q);
063    return buildStringQuery(qValue);
064  }
065
066  /**
067   * Helper method that sets the parameters for full text search param "q".
068   * It will also add the optional query function.
069   */
070  @Override
071  protected void initTemplates(Class<?> annotatedClass) {
072    if (!annotatedClass.isAnnotationPresent(SuggestMapping.class)) {
073      throw new IllegalStateException(
074        "Given class " + annotatedClass.getCanonicalName() + " is missing the SuggestField annotation");
075    }
076    SuggestMapping annotation = annotatedClass.getAnnotation(SuggestMapping.class);
077    String phraseQueryField = annotation.phraseQueryField();
078    String field = annotation.field();
079    String phraseQuery = PARAMS_JOINER.join(phraseQueryField, toPhraseQuery(QUERY_PLACE_HOLDER));
080    phraseQueryTemplate = toBoostedQuery(phraseQuery, PHRASE_QUERY_BOOST, false);
081    startPhraseQueryTemplate =
082      toBoostedQuery(PARAMS_JOINER.join(phraseQueryField, QUERY_PLACE_HOLDER), PHRASE_QUERY_BOOST, false);
083    queryTemplate = PARAMS_JOINER.join(field, QUERY_PLACE_HOLDER);
084  }
085
086  /**
087   * Creates a query string that matches terms by to the partial phrase query fields.
088   * The method takes a query q with the form: "pum conc" and produces a query with the form, for input puma con:
089   * "(canonical_name_auto:\"puma con\"^1000 OR canonical_name_auto:puma^1000) OR
090   * (canonical_name:puma^300 OR canonical_name:con^200)"
091   *
092   * @param q input query
093   * @return a string containing a valid Solr query
094   */
095  private String buildStringQuery(String q) {
096    String cleanQ = clearConsecutiveBlanks(q);
097    String[] tokens = cleanQ.split("\\ ");
098    Collection<String> query = new ArrayList<String>();
099    Collection<String> partialQuery = new ArrayList<String>();
100    Integer partialBoost = PARTIAL_QUERY_BOOST;
101    if (tokens.length > 1) {
102      String[] phraseQuery = new String[2];
103      phraseQuery[0] = phraseQueryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(cleanQ));
104      phraseQuery[1] = startPhraseQueryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(tokens[0]));
105      query.add(toParenthesesQuery(PARAMS_OR_JOINER.join(phraseQuery)));
106      for (String token : tokens) {
107        partialQuery.add(toBoostedQuery(queryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(token)),
108            partialBoost, false));
109        if (partialBoost > PARTIAL_BOOST_DECREMENT) {
110          partialBoost -= PARTIAL_BOOST_DECREMENT;
111        }
112      }
113    } else {
114      if (SolrConstants.DEFAULT_FILTER_QUERY.equals(cleanQ)) {
115        query.add(toParenthesesQuery(startPhraseQueryTemplate.replace(QUERY_PLACE_HOLDER, cleanQ)));
116        partialQuery.add(toBoostedQuery(queryTemplate.replace(QUERY_PLACE_HOLDER, tokens[0]), partialBoost, false));
117      } else {
118        query.add(toParenthesesQuery(startPhraseQueryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(cleanQ))));
119        partialQuery.add(toBoostedQuery(queryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(tokens[0])), partialBoost, false));
120      }
121
122
123    }
124    query.add(toParenthesesQuery(PARAMS_OR_JOINER.join(partialQuery)));
125    return PARAMS_OR_JOINER.join(query);
126  }
127}
