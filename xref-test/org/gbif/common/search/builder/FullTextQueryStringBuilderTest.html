22    private final FullTextQueryStringBuilder queryStringBuilder = FullTextQueryStringBuilder
23      .create(TestSearchResult.class);
24    private final String EXPECTED_PUMA_QUERY =
25      "(canonical_name:puma^100.0 OR class:puma^1.0 OR description_ft:puma^0.3 OR family:puma^2.0 OR genus:puma^2.0 OR kingdom:puma^1.0 OR order:puma^1.0 OR phylum:puma^1.0 OR scientific_name_ft:puma^10.0 OR scientific_name_ft:*puma*^0.2 OR species:puma^5.0 OR subgenus:puma^3.0 OR vernacular_name_ft:puma^6.0 OR vernacular_name_ft:*puma*^0.2)";

27    @Test
28    public void buildFullTextTest() {
29      String query = queryStringBuilder.build("puma");
30      assertTrue(query.contains("*"));
31    }

34    @Test
35    public void escapeSpecialChars() {
36      FacetedSearchRequest<TestSearchParameter> searchRequest = new FacetedSearchRequest<TestSearchParameter>();
37      searchRequest.addFacets(TestSearchParameter.values());
38      searchRequest.setMultiSelectFacets(true);
39      searchRequest.setQ("puma-con");
40      SolrQuery solrQuery = buildQuery(searchRequest);
41      assertEquals(EXPECTED_PUMA_QUERY.replaceAll("puma", "puma\\\\-con"), solrQuery.getQuery());
42    }

72    @Test
73    public void buildTagFacetParamTest() {
74      String result = SolrConstants.FACET_FILTER_EX.replace(SolrConstants.TAG_FIELD_PARAM, "field");
75      String resultWrong =
76        SolrConstants.FACET_FILTER_EX.replace(SolrConstants.TAG_FIELD_PARAM, "field").replace("{", " ");
77      Matcher matcherOK = SolrConstants.FACET_FILTER_RGEX.matcher(result);
78      Matcher matcherError = SolrConstants.FACET_FILTER_RGEX.matcher(resultWrong);
79      Matcher matcherCleaner = SolrConstants.FACET_FILTER_RGEX_CLEAN.matcher(result);
80      assertTrue(matcherOK.matches());
81      assertFalse(matcherError.matches());
82      assertEquals("field", matcherCleaner.replaceAll(""));
83      assertNotNull(result);
84    }

96    private SolrQuery buildQuery(FacetedSearchRequest<TestSearchParameter> searchRequest) {
97      SolrQueryBuilder<TestSearchResult, TestSearchParameter> requestBuilder =
98        SolrQueryBuilder.create(TestSearchParameter.class, TestSearchResult.class);
99      requestBuilder.withQueryBuilder(queryStringBuilder);
100     return requestBuilder.build(searchRequest);
101   }
