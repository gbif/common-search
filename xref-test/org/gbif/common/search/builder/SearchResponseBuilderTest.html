18  public class SearchResponseBuilderTest {
19  
20    /**
21     * Tests that returned ordinal facet values are converted to actual enum names.
22     */
23    @Test
24    public void testBuild() throws Exception {
25      final SearchResponseBuilder builder =
26        SearchResponseBuilder.create(TestSearchResult.class, TestSearchResult.class, TestSearchParameter.class);
27  
28      PagingRequest req = new PagingRequest(0,5);
29      QueryResponse queryResponse = mock(QueryResponse.class);
30      SolrDocumentList solrList = new SolrDocumentList();
31      when(queryResponse.getResults()).thenReturn(solrList);
32      List<FacetField> facets = Lists.newArrayList();
33      FacetField rankFacet = new FacetField("rank");
34      rankFacet.add("19", 87); // genus
35      rankFacet.add("14", 14); // family
36      rankFacet.add("1", 8);  // kingdom
37      facets.add(rankFacet);
38      when(queryResponse.getFacetFields()).thenReturn(facets);
39  
40      SearchResponse<TestSearchResult, TestSearchParameter> resp = builder.build(req, queryResponse);
41  
42      assertEquals(1, resp.getFacets().size());
43      assertEquals(TestSearchParameter.RANK, resp.getFacets().get(0).getField());
44      assertEquals(3, resp.getFacets().get(0).getCounts().size());
45      assertEquals("GENUS", resp.getFacets().get(0).getCounts().get(0).getName() );
46      assertEquals(87l, resp.getFacets().get(0).getCounts().get(0).getCount().longValue());
47      assertEquals("FAMILY", resp.getFacets().get(0).getCounts().get(1).getName() );
48      assertEquals(14l, resp.getFacets().get(0).getCounts().get(1).getCount().longValue());
49      assertEquals("KINGDOM", resp.getFacets().get(0).getCounts().get(2).getName() );
50      assertEquals(8l, resp.getFacets().get(0).getCounts().get(2).getCount().longValue());
51    }
