30    @Test
31    public void buildSearchRequestCombined() {
32      TestSearchRequest req = new TestSearchRequest();
33      req.addParameter(TestSearchParameter.HIGHERTAXON, "111");
34      req.addParameter(TestSearchParameter.RANK, Rank.SPECIES);
35      req.addParameter(TestSearchParameter.RANK, Rank.VARIETY);
36      req.addParameter(TestSearchParameter.EXTINCT, true);
37      req.addFacets(TestSearchParameter.HIGHERTAXON, TestSearchParameter.RANK);
38  
39      SolrQuery query = buildQuery(req);
40  
41      Set<String> facets = Sets.newHashSet(query.getFacetFields());
42      assertEquals(2, facets.size());
43      assertTrue(facets.contains("rank"));
44      assertTrue(facets.contains("higher_taxon_nub_key"));
45      assertFilterQueryContains(query, "{!tag=ffqextinct}extinct:true");
46      assertFilterQueryContains(query, "{!tag=ffqrank}(rank:26 OR rank:30)");
47      assertFilterQueryContains(query, "{!tag=ffqhigher_taxon_nub_key}higher_taxon_nub_key:111");
48    }

61    @Test
62    public void buildSearchRequestFaceted() {
63      final UUID uuid1 = UUID.randomUUID();
64      final UUID uuid2 = UUID.randomUUID();
65  
66      FacetedSearchRequest<TestSearchParameter> searchRequest = new FacetedSearchRequest<TestSearchParameter>();
67      searchRequest.setQ("puma");
68      searchRequest.addFacets(TestSearchParameter.values());
69      searchRequest.setMultiSelectFacets(true);
70      searchRequest.addParameter(TestSearchParameter.CHECKLIST, uuid1.toString());
71      searchRequest.addParameter(TestSearchParameter.CHECKLIST, uuid2.toString());
72  
73      SolrQuery query = buildQuery(searchRequest);
74      assertNotNull(query.getFilterQueries());
75      assertNotNull(query.getFacetFields());
76      assertEquals(1, query.getFilterQueries().length);
77      final String x = query.getFilterQueries()[0];
78      assertTrue(x.contains("checklist_key:" + uuid1.toString()));
79      assertTrue(x.contains("checklist_key:" + uuid2.toString()));
80      // the search example doesnt have THREAT mapped as a facet, only regular param
81      assertEquals(TestSearchParameter.values().length - 1, query.getFacetFields().length);
82    }

154   private void assertFilterQueryContains(SolrQuery query, String filter) {
155     for (String fq : query.getFilterQueries()) {
156       if (filter.equals(fq)) {
157         return;
158       }
159     }
160     fail("Filter query does not contain " + filter + " : " + Joiner.on(" ").join(query.getFilterQueries()));
161   }
