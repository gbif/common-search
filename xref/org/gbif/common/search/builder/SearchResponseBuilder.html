50  /**
51   * Builder class that helps in the creation process of {@link SearchResponse} objects from
52   * {@link org.apache.solr.client.solrj.response.QueryResponse} objects.
53   * The build method is not thread safe, supports the cloneable interface for allowing holds a partial state,
54   * that can be cloned and repeated in other instances.
55   *
56   * @param <T> is the response type.
57   * @param <ST> is the type of the annotated class that holds the mapping information.
58   * @param <P> is the search parameter enumeration.
59   */
60  public class SearchResponseBuilder<T, ST extends T, P extends Enum<?> & SearchParameter> {
61  
62    // Logger
63    private static final Logger LOG = LoggerFactory.getLogger(SearchResponseBuilder.class);
64  
65    // Holds a map containing the solr fields vs. application facet names
66    private final BiMap<String, P> solrField2ParamEnumMap;
67  
68    // Holds a map containing the solr fields vs. java fields names
69    private final BiMap<String, String> solrField2javaPropertiesMap;
70  
71    // Holds a map containing the highlightable solr fields vs. java fields names
72    private Map<String, String> hlFieldPropertyPropertiesMap;
73  
74    // Name of the field annotated with the Key annotation.
75    private final String keyField;
76  
77  
78    // Class that contains the annotations for creating the response objects.
79    private final Class<T> responseClass;
80    private final Class<ST> annotatedClass;
81  
82    /**
83     * Private full constructor of final fields only.
84     */
85    private SearchResponseBuilder(Class<T> responseClass, Class<ST> annotatedClass,
86                                  BiMap<String, P> solrField2ParamEnumMap,
87                                  BiMap<String, String> solrField2javaPropertiesMap) {
88      this.responseClass = responseClass;
89      this.annotatedClass = annotatedClass;
90      this.solrField2ParamEnumMap = solrField2ParamEnumMap;
91      this.solrField2javaPropertiesMap = solrField2javaPropertiesMap;
92      keyField = getKeyField(annotatedClass);
93    }
94  
95    /**
96     * Private default constructor.
97     */
98    private SearchResponseBuilder(Class<T> responseClass, Class<ST> annotatedClass, Class<P> enumClass) {
99      this.responseClass = responseClass;
100     this.annotatedClass = annotatedClass;
101     solrField2ParamEnumMap = initFacetFieldsPropertiesMap(annotatedClass, enumClass);
102     solrField2javaPropertiesMap = initFieldsPropertiesMap(annotatedClass);
103     keyField = getKeyField(annotatedClass);
104   }

134   /**
135    * Builds a SearchResponse instance using the current builder state.
136    *
137    * @return a new instance of a SearchResponse.
138    */
139   public SearchResponse<T, P> build(Pageable searchRequest, QueryResponse queryResponse) {
140     // Create response
141     SearchResponse<T, P> searchResponse = new SearchResponse<T, P>(searchRequest);
142     searchResponse.setCount(queryResponse.getResults().getNumFound());
143     searchResponse.setLimit(queryResponse.getResults().size());
144     // The results and facets are copied into the response
145     final List<ST> resultsST = queryResponse.getBeans(annotatedClass);
146     // convert types
147     final List<T> results = Lists.newArrayList();
148     for (ST x : resultsST) {
149       results.add(x);
150     }
151     searchResponse.setResults(results);
152     searchResponse.setFacets(getFacetsFromResponse(queryResponse));
153     setHighlightedFields(searchResponse, queryResponse);
154     return searchResponse;
155   }

158    * Builds a simple SearchResponse instance using the current builder state, facets and highlighting reponses are
159    * ommited.

161    * @return a new instance of a SearchResponse.

163   public static <T, ST extends T, P extends SearchParameter> SearchResponse<T, P> buildSuggestReponse(
164       Pageable searchRequest, QueryResponse queryResponse, Class<ST> annotatedClass) {
165     // Create response
166     SearchResponse<T, P> searchResponse = new SearchResponse<T, P>(searchRequest);
167     searchResponse.setCount(queryResponse.getResults().getNumFound());
168     searchResponse.setLimit(queryResponse.getResults().size());
169     // The results and facets are copied into the response
170     final List<ST> resultsST = queryResponse.getBeans(annotatedClass);
171     // convert types
172     final List<T> results = Lists.newArrayList();
173     for (ST x : resultsST) {
174       results.add(x);
175     }
176     searchResponse.setResults(results);
177     return searchResponse;
178   }

193   /**
194    * Sets the list of highlighted fields.
195    */
196   public SearchResponseBuilder<T, ST, P> withHighlightFields(List<String> highlightFields) {
197     hlFieldPropertyPropertiesMap = new HashMap<String, String>();
198     for (String hlField : highlightFields) {
199       if (solrField2javaPropertiesMap.containsKey(hlField)) {
200         hlFieldPropertyPropertiesMap.put(hlField, solrField2javaPropertiesMap.get(hlField));
201       }
202     }
203     return this;
204   }

206   /**
207    * Cleans all occurrences of highlighted tags/marks in the parameter and returns an new instance clean of those
208    * marks.
209    */
210   private String cleanHighlightingMarks(final String hlText) {
211     String hlLiteral = hlText;
212     int indexPre = hlLiteral.indexOf(HL_PRE);
213     while (indexPre > -1) {
214       int indexPost = hlLiteral.indexOf(HL_POST, indexPre + HL_PRE.length());
215       if (indexPost > -1) {
216         String post = hlLiteral.substring(indexPost + HL_POST.length());
217         String pre = hlLiteral.substring(0, indexPost);
218         Matcher preMatcher = HL_PRE_REGEX.matcher(pre);
219         pre = preMatcher.replaceFirst("");
220         hlLiteral = pre + post;
221       }
222       indexPre = hlLiteral.indexOf(HL_PRE);
223     }
224     return hlLiteral;
225   }

228   private T getByKey(SearchResponse<T, P> response, String key) {
229     for (T bean : response.getResults()) {
230       try {
231         String value = BeanUtils.getProperty(bean, keyField);
232         if (value.equals(key)) {
233           return bean;
234         }
235       } catch (IllegalAccessException e) {
236         LOG.error("Error accesing key field", e);
237         throw new SearchException(e);
238       } catch (InvocationTargetException e) {
239         LOG.error("Error invoking set key field", e);
240         throw new SearchException(e);
241       } catch (NoSuchMethodException e) {
242         LOG.error("Error invoking setter method", e);
243         throw new SearchException(e);
244       }
245     }
246     return null;
247   }

249   /**
250    * Helper method that takes Solr response and extracts the facets results.
251    * The facets are converted to a list of Facets understood by the search API.
252    * The result of this method can be a empty list.
253    *
254    * @param queryResponse that contains the facets information returned by Solr
255    * @return the List of facets retrieved from the Solr response
256    */
257   private List<Facet<P>> getFacetsFromResponse(final QueryResponse queryResponse) {
258     List<Facet<P>> facets = Lists.newArrayList();
259     if (queryResponse.getFacetFields() != null) {
260       for (final FacetField facetField : queryResponse.getFacetFields()) {
261         P facetParam = solrField2ParamEnumMap.get(facetField.getName());
262         Facet<P> facet = new Facet<P>(facetParam);
263 
264         List<Facet.Count> counts = Lists.newArrayList();
265         if (facetField.getValues() != null) {
266           for (final FacetField.Count count : facetField.getValues()) {
267             String value = count.getName();
268             if (Enum.class.isAssignableFrom(facetParam.type())) {
269               value = getFacetEnumValue(facetParam, value);
270             }
271             counts.add(new Facet.Count(value, count.getCount()));
272           }
273         }
274         facet.setCounts(counts);
275         facets.add(facet);
276       }
277     }
278     return facets;
279   }

282   /**
283    * Gets the facet value of Enum type parameter.
284    * If the Enum is either a Country or a Language, its iso2Letter code it's used.
285    */
286   private String getFacetEnumValue(P facetParam, String value) {
287     // the expected enum type for the value if it is an enum - otherwise null
288     final Enum<?>[] enumValues = ((Class<? extends Enum<?>>) facetParam.type()).getEnumConstants();
289     // if we find integers these are ordinals, translate back to enum names
290     final Integer intValue = Ints.tryParse(value);
291     if (intValue != null) {
292       final Enum<?> enumValue = enumValues[intValue];
293       if (Country.class.equals(facetParam.type())) {
294         return ((Country) enumValue).getIso2LetterCode();
295       } else if (Language.class.equals(facetParam.type())) {
296         return ((Language) enumValue).getIso2LetterCode();
297       } else {
298         return enumValue.name();
299       }
300     } else {
301       if (Country.class.equals(facetParam.type())) {
302         return Country.fromIsoCode(value).getIso2LetterCode();
303       } else if (Language.class.equals(facetParam.type())) {
304         return Language.fromIsoCode(value).getIso2LetterCode();
305       } else {
306         return VocabularyUtils.lookupEnum(value, (Class<? extends Enum<?>>) facetParam.type()).name();
307       }
308     }
309   }

311   /**
312    * Takes the highlighted fields form solrResponse and copies them to the response object.
313    *
314    * @param response to set the highlighted fields.
315    * @param solrResponse to extract the highlighting information
316    */
317   private void setHighlightedFields(SearchResponse<T, P> response, final QueryResponse solrResponse) {
318     if ((this.hlFieldPropertyPropertiesMap != null) && (solrResponse.getHighlighting() != null) && !solrResponse
319       .getHighlighting().isEmpty()) {
320       for (String docId : solrResponse.getHighlighting().keySet()) {
321         setHighlightedFieldValues(response, solrResponse, docId);
322       }
323     }
324   }

326   /**
327    * Sets the value of the highlighted field in object 'bean'.
328    */
329   private void setHighlightedFieldValue(T bean, String solrField, String hlValue) {
330     try {
331       if (List.class
332         .isAssignableFrom(PropertyUtils.getPropertyType(bean, solrField2javaPropertiesMap.get(solrField)))) {
333         List<String> listProperty =
334           (List<String>) PropertyUtils.getProperty(bean, solrField2javaPropertiesMap.get(solrField));
335         // Cleans the hl markers
336         String hlCleanValue = cleanHighlightingMarks(hlValue);
337         // Position of the value
338         int hlValueIndex = listProperty.indexOf(hlCleanValue);
339         if (hlValueIndex != -1) { // replace the value with the highlighted value
340           listProperty.remove(hlValueIndex);
341           listProperty.add(hlValueIndex, hlValue);
342         }
343 
344       } else if (HighlightableList.class
345         .isAssignableFrom(PropertyUtils.getPropertyType(bean, solrField2javaPropertiesMap.get(solrField)))) {
346         HighlightableList property =
347           (HighlightableList) PropertyUtils.getProperty(bean, solrField2javaPropertiesMap.get(solrField));
348         // Cleans the hl markers
349         String hlCleanValue = cleanHighlightingMarks(hlValue);
350         // Position of the value
351         List<String> values = property.getValues();
352         int hlValueIndex = values.indexOf(hlCleanValue);
353         if (hlValueIndex != -1) { // replace the value with the highlighted value
354           property.replaceValue(hlValueIndex, hlValue);
355         }
356 
357       } else {
358         String propertyName = solrField2javaPropertiesMap.get(solrField);
359         if (PropertyUtils.getPropertyDescriptor(bean, propertyName).getPropertyType().isAssignableFrom(String.class)) {
360           BeanUtils.setProperty(bean, propertyName, hlValue);
361         }
362       }
363     } catch (IllegalAccessException e) {
364       LOG.error("Error accessing field for setting highlighted value", e);
365       throw new SearchException(e);
366     } catch (InvocationTargetException e) {
367       LOG.error("Error setting highlighted value", e);
368       throw new SearchException(e);
369     } catch (NoSuchMethodException e) {
370       LOG.error("Error invoking method to set highlighted value", e);
371       throw new SearchException(e);
372     }
373   }

375   /**
376    * Sets the highlighted field values of beans in the response object.
377    */
378   private void setHighlightedFieldValues(SearchResponse<T, P> response, final QueryResponse solrResponse,
379     String docId) {
380     T bean = getByKey(response, docId);
381     if (bean != null) {
382       Map<String, List<String>> docHighlights = solrResponse.getHighlighting().get(docId);
383       for (String solrField : hlFieldPropertyPropertiesMap.keySet()) {
384         if (docHighlights.containsKey(solrField)) {
385           for (String hlValue : docHighlights.get(solrField)) { // get each snippet
386             setHighlightedFieldValue(bean, solrField, hlValue);
387           }
388         }
389       }
390     }
391   }
