13  package org.gbif.common.search.builder;
14  
15  import org.gbif.common.search.model.SuggestMapping;
16  import org.gbif.common.search.util.QueryUtils;
17  import org.gbif.common.search.util.SolrConstants;
18  
19  import java.util.ArrayList;
20  import java.util.Collection;
21  import org.apache.solr.client.solrj.util.ClientUtils;
22  
23  import static org.gbif.common.search.util.QueryUtils.PARAMS_JOINER;
24  import static org.gbif.common.search.util.QueryUtils.PARAMS_OR_JOINER;
25  import static org.gbif.common.search.util.QueryUtils.clearConsecutiveBlanks;
26  import static org.gbif.common.search.util.QueryUtils.toBoostedQuery;
27  import static org.gbif.common.search.util.QueryUtils.toParenthesesQuery;
28  import static org.gbif.common.search.util.QueryUtils.toPhraseQuery;
29  
30  /**
31   * Builder class that helps in the creation process of query patterns for classes annotated with
32   * {@link org.gbif.common.search.model.SuggestMapping}.
33   */
34  public class SuggestQueryStringBuilder extends QueryStringBuilderBase {
35  
36    private static final Integer PHRASE_QUERY_BOOST = 1000;
37  
38    private static final Integer PARTIAL_QUERY_BOOST = 300;
39  
40    private static final Integer PARTIAL_BOOST_DECREMENT = 100;
41  
42    private String startPhraseQueryTemplate;
43  
44    /**
45     * Private default constructor.
46     */
47    private SuggestQueryStringBuilder(Class<?> annotatedClass) {
48      super(annotatedClass);
49    }
50  
51    public static SuggestQueryStringBuilder create(Class<?> annotatedClass) {
52      return new SuggestQueryStringBuilder(annotatedClass);
53    }
54  
55  
56    /*
57     * (non-Javadoc)
58     * @see org.gbif.common.search.builder.QueryStringBuilderBase#build(java.lang.String)
59     */
60    @Override
61    public String build(String q) {
62      final String qValue = QueryUtils.emptyToDefaultQuery(q);
63      return buildStringQuery(qValue);
64    }
65  
66    /**
67     * Helper method that sets the parameters for full text search param "q".
68     * It will also add the optional query function.
69     */
70    @Override
71    protected void initTemplates(Class<?> annotatedClass) {
72      if (!annotatedClass.isAnnotationPresent(SuggestMapping.class)) {
73        throw new IllegalStateException(
74          "Given class " + annotatedClass.getCanonicalName() + " is missing the SuggestField annotation");
75      }
76      SuggestMapping annotation = annotatedClass.getAnnotation(SuggestMapping.class);
77      String phraseQueryField = annotation.phraseQueryField();
78      String field = annotation.field();
79      String phraseQuery = PARAMS_JOINER.join(phraseQueryField, toPhraseQuery(QUERY_PLACE_HOLDER));
80      phraseQueryTemplate = toBoostedQuery(phraseQuery, PHRASE_QUERY_BOOST, false);
81      startPhraseQueryTemplate =
82        toBoostedQuery(PARAMS_JOINER.join(phraseQueryField, QUERY_PLACE_HOLDER), PHRASE_QUERY_BOOST, false);
83      queryTemplate = PARAMS_JOINER.join(field, QUERY_PLACE_HOLDER);
84    }
85  
86    /**
87     * Creates a query string that matches terms by to the partial phrase query fields.
88     * The method takes a query q with the form: "pum conc" and produces a query with the form, for input puma con:
89     * "(canonical_name_auto:\"puma con\"^1000 OR canonical_name_auto:puma^1000) OR
90     * (canonical_name:puma^300 OR canonical_name:con^200)"
91     *
92     * @param q input query
93     * @return a string containing a valid Solr query
94     */
95    private String buildStringQuery(String q) {
96      String cleanQ = clearConsecutiveBlanks(q);
97      String[] tokens = cleanQ.split("\\ ");
98      Collection<String> query = new ArrayList<String>();
99      Collection<String> partialQuery = new ArrayList<String>();
100     Integer partialBoost = PARTIAL_QUERY_BOOST;
101     if (tokens.length > 1) {
102       String[] phraseQuery = new String[2];
103       phraseQuery[0] = phraseQueryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(cleanQ));
104       phraseQuery[1] = startPhraseQueryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(tokens[0]));
105       query.add(toParenthesesQuery(PARAMS_OR_JOINER.join(phraseQuery)));
106       for (String token : tokens) {
107         partialQuery.add(toBoostedQuery(queryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(token)),
108             partialBoost, false));
109         if (partialBoost > PARTIAL_BOOST_DECREMENT) {
110           partialBoost -= PARTIAL_BOOST_DECREMENT;
111         }
112       }
113     } else {
114       if (SolrConstants.DEFAULT_FILTER_QUERY.equals(cleanQ)) {
115         query.add(toParenthesesQuery(startPhraseQueryTemplate.replace(QUERY_PLACE_HOLDER, cleanQ)));
116         partialQuery.add(toBoostedQuery(queryTemplate.replace(QUERY_PLACE_HOLDER, tokens[0]), partialBoost, false));
117       } else {
118         query.add(toParenthesesQuery(startPhraseQueryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(cleanQ))));
119         partialQuery.add(toBoostedQuery(queryTemplate.replace(QUERY_PLACE_HOLDER, ClientUtils.escapeQueryChars(tokens[0])), partialBoost, false));
120       }
121 
122 
123     }
124     query.add(toParenthesesQuery(PARAMS_OR_JOINER.join(partialQuery)));
125     return PARAMS_OR_JOINER.join(query);
126   }
127 }
