13  package org.gbif.common.search.builder;
14  
15  import org.gbif.common.search.model.FullTextSearchField;
16  import org.gbif.common.search.model.WildcardPadding;
17  
18  import java.util.List;
19  
20  import com.google.common.collect.Lists;
21  
22  
23  import static org.gbif.common.search.util.AnnotationUtils.initFullTextFieldsPropertiesMap;
24  import static org.gbif.common.search.util.QueryUtils.PARAMS_JOINER;
25  import static org.gbif.common.search.util.QueryUtils.getExactMatchField;
26  import static org.gbif.common.search.util.QueryUtils.getMatchPatternAndScore;
27  import static org.gbif.common.search.util.SolrConstants.SCORE_OP;
28  
29  
30  /**
31   * Builder class that helps in the creation process of query patterns for classes annotated with
32   * {@link FullTextSearchField}.
33   */
34  public class FullTextQueryStringBuilder extends QueryStringBuilderBase {
35  
36    /**
37     * Private default constructor.
38     */
39    private FullTextQueryStringBuilder(Class<?> annotatedClass) {
40      super(annotatedClass);
41    }
42  
43    public static FullTextQueryStringBuilder create(Class<?> annotatedClass) {
44      return new FullTextQueryStringBuilder(annotatedClass);
45    }
46  
47  
48    /**
49     * Helper method that sets the parameters for full text search param "q".
50     * It will also add the optional query function.
51     */
52    @Override
53    protected void initTemplates(Class<?> annotatedClass) {
54      List<String> fullTextQueryComponents = Lists.newArrayList();
55      List<String> phraseQueryComponents = Lists.newArrayList();
56  
57      List<FullTextSearchField> fullTextSearchFields = initFullTextFieldsPropertiesMap(annotatedClass);
58      if (!fullTextSearchFields.isEmpty()) {
59        for (FullTextSearchField field : fullTextSearchFields) {
60          String exactMatchField = getExactMatchField(field);
61          String exactMatchSubQuery =
62              PARAMS_JOINER.join(exactMatchField, QUERY_PLACE_HOLDER + SCORE_OP + field.exactMatchScore());
63          fullTextQueryComponents.add(exactMatchSubQuery);
64          phraseQueryComponents.add(exactMatchSubQuery);
65          setWildcardFilters(fullTextQueryComponents, phraseQueryComponents, field, exactMatchField);
66          highlightedFields.add(field.highlightField().isEmpty() ? field.field() : field.highlightField());
67        }
68        queryTemplate = buildQueryExpression(fullTextQueryComponents);
69        phraseQueryTemplate = buildQueryExpression(phraseQueryComponents);
70      }
71    }
72  
73  
74    /**
75     * Sets the filters for wildcard (if applicable) queries.
76     */
77    private void setWildcardFilters(List<String> fullTextQueryComponents, List<String> phraseQueryComponents,
78                                    FullTextSearchField field, String exactMatchField) {
79      if (field.partialMatching() != WildcardPadding.NONE) {
80        fullTextQueryComponents
81          .add(PARAMS_JOINER.join(field.field(), getMatchPatternAndScore(field, QUERY_PLACE_HOLDER)));
82        if (!exactMatchField.equals(field.field())) {
83          phraseQueryComponents
84            .add(PARAMS_JOINER.join(field.field(), QUERY_PLACE_HOLDER + SCORE_OP + field.partialMatchScore()));
85        }
86      }
87    }
88  }
