58  /**
59   * Builder class that helps in the creation process of {@link SolrQuery} objects from {@link SearchRequest} objects.
60   * The build method is not thread safe, supports the cloneable interface for allowing holds a partial state,
61   * that can be cloned and repeated in other instances.
62   *
63   * @param <T> is the type of the annotated class that holds the mapping information.
64   * @param <P> is the SearchParameter Enum type used for enumerate the supported facets / search parameter.
65   */
66  public class SolrQueryBuilder<T, P extends Enum<?> & SearchParameter> {
67  
68    private static final Logger LOG = LoggerFactory.getLogger(SolrQueryBuilder.class);
69    private static final FacetField.SortOrder DEFAULT_FACET_SORT = FacetField.SortOrder.COUNT;
70    private static final int DEFAULT_FACET_COUNT = 1;
71    private static final boolean DEFAULT_FACET_MISSING = true;
72    private QueryStringBuilderBase queryBuilder;
73  
74    // Request request handler
75    private String requestHandler;
76    // Search parameter/facet enumeration
77    private final Class<P> searchParameterEnum;
78    // a map containing the solr fields vs. facet enum members
79    private final BiMap<String, P> facetFieldsPropertiesMap;
80    // keyed on solr field names
81    private final Map<String, FacetField> facetFieldDefs;
82    // a map containing the solr fields vs. property name
83    private final BiMap<String, String> fieldsPropertiesMap;
84    // the value used for an optional sort order applied to a search via param "sort"
85    private Map<String, SolrQuery.ORDER> primarySortOrder;
86  
87    /**
88     * Full private constructor.
89     */
90    private SolrQueryBuilder(Class<P> searchParameterEnum, BiMap<String, String> fieldsPropertiesMap,
91        BiMap<String, P> facetFieldsPropertiesMap, Map<String, FacetField> facetFieldDefs) {
92      this.searchParameterEnum = searchParameterEnum;
93      this.facetFieldsPropertiesMap = facetFieldsPropertiesMap;
94      this.fieldsPropertiesMap = fieldsPropertiesMap;
95      this.facetFieldDefs = facetFieldDefs;
96    }
97  
98    /**
99     * Default private constructor.
100    */
101   private SolrQueryBuilder(Class<P> searchParameterEnum, Class<T> annotatedClass) {
102     this.searchParameterEnum = searchParameterEnum;
103     this.facetFieldsPropertiesMap = initFacetFieldsPropertiesMap(annotatedClass, searchParameterEnum);
104     this.fieldsPropertiesMap = initFieldsPropertiesMap(annotatedClass);
105     this.facetFieldDefs = initFacetFieldDefs(annotatedClass);
106   }

108   /**
109    * Default factory method.
110    */
111   public static <T, P extends Enum<?> & SearchParameter> SolrQueryBuilder<T, P> create(Class<P> searchParameterEnum,
112       Class<T> annotatedClass) {
113     return new SolrQueryBuilder<T, P>(searchParameterEnum, annotatedClass);
114   }

116   /**
117    * Builds a SolrQuery instance using the searchRequest.
118    */
119   public SolrQuery build(final FacetedSearchRequest<P> searchRequest) {
120     // get regular search without facets
121     SolrQuery solrQuery = build((SearchRequest<P>) searchRequest);
122     // add facets
123     applyFacetSettings(searchRequest, solrQuery);
124 
125     LOG.debug("Solr faceted query build: {}", solrQuery);
126     return solrQuery;
127   }
128 
129   public SolrQuery build(final SearchRequest<P> searchRequest) {
130     SolrQuery solrQuery = new SolrQuery();
131     // q param
132     solrQuery.setQuery(queryBuilder.build(searchRequest));
133     // paging
134     setQueryPaging(searchRequest, solrQuery);
135     // sets the default, alternative query
136     solrQuery.set(ALT_QUERY_PARAM, DEFAULT_QUERY);
137     // filters
138     setQueryFilter(searchRequest, solrQuery);
139     // highlight
140     setHighLightParams(searchRequest, solrQuery);
141     // sorting
142     setPrimarySortOrder(solrQuery);
143     // set shards/for distributed search
144     setRequestHandler(solrQuery, requestHandler);
145 
146     LOG.debug("Solr query build: {}", solrQuery);
147     return solrQuery;
148   }

150   /**

152    */
153   public SolrQueryBuilder<T, P> getCopy() {
154     SolrQueryBuilder<T, P> searchRequestBuilder =
155         new SolrQueryBuilder<T, P>(searchParameterEnum, fieldsPropertiesMap, facetFieldsPropertiesMap, facetFieldDefs);
156     searchRequestBuilder.queryBuilder = queryBuilder;
157     searchRequestBuilder.primarySortOrder = primarySortOrder;
158     searchRequestBuilder.requestHandler = requestHandler;
159     return searchRequestBuilder;
160   }

196   /**
197    * Helper method that sets the parameter for a faceted query.
198    *
199    * @param searchRequest the searchRequest used to extract the parameters
200    * @param solrQuery this object is modified by adding the facets parameters
201    */
202   private void applyFacetSettings(FacetedSearchRequest<P> searchRequest, SolrQuery solrQuery) {
203 
204     final BiMap<P, String> facetFieldsMapInv = facetFieldsPropertiesMap.inverse();
205 
206     if (!searchRequest.getFacets().isEmpty()) {
207       // Only show facets that contains at least 1 record
208       solrQuery.setFacet(true);
209       // defaults if not overridden on per field basis
210       solrQuery.setFacetMinCount(MoreObjects.firstNonNull(searchRequest.getFacetMinCount(), DEFAULT_FACET_COUNT));
211       solrQuery.setFacetMissing(DEFAULT_FACET_MISSING);
212       solrQuery.setFacetSort(DEFAULT_FACET_SORT.toString().toLowerCase());
213 
214       for (final P facet : searchRequest.getFacets()) {
215         if (!facetFieldsMapInv.containsKey(facet)) {
216           LOG.warn("{} is no valid facet. Ignore", facet);
217           continue;
218         }
219         final String field = facetFieldsMapInv.get(facet);
220         if (searchRequest.isMultiSelectFacets()) {
221           // use exclusion filter with same name as used in filter query
222           // http://wiki.apache.org/solr/SimpleFacetParameters#Tagging_and_excluding_Filters
223           solrQuery.addFacetField(FACET_FILTER_EX.replace(TAG_FIELD_PARAM, field));
224         } else {
225           solrQuery.addFacetField(field);
226         }
227         FacetField fieldDef = facetFieldDefs.get(field);
228         if (fieldDef.missing() != DEFAULT_FACET_MISSING) {
229           solrQuery.setParam(perFieldParamName(field, PARAM_FACET_MISSING), fieldDef.missing());
230         }
231         if (fieldDef.sort() != DEFAULT_FACET_SORT) {
232           solrQuery.setParam(perFieldParamName(field, PARAM_FACET_SORT), fieldDef.sort().toString().toLowerCase());
233         }
234         setFacetMethod(solrQuery, field, fieldDef.method());
235       }
236     }
237   }

239   /**
240    * Utility method that creates the resulting Solr expression for facet and general query filters parameters.
241    */
242   private StringBuilder buildFilterQuery(final boolean isFacetedRequest, final String solrFieldName,
243       List<String> filterQueriesComponents) {
244     StringBuilder filterQuery = new StringBuilder();
245     if (isFacetedRequest) {
246       filterQuery.append(FACET_FILTER_TAG.replace(TAG_FIELD_PARAM, solrFieldName));
247     }
248     if (filterQueriesComponents.size() > 1) {
249       filterQuery.append('(');
250       filterQuery.append(PARAMS_OR_JOINER.join(filterQueriesComponents));
251       filterQuery.append(')');
252     } else {
253       filterQuery.append(PARAMS_OR_JOINER.join(filterQueriesComponents));
254     }
255     return filterQuery;
256   }

258   /**
259    * Interprets the value of parameter "value" using types pType (Parameter type) and eType (Enumeration).
260    */
261   private static String getInterpretedValue(final Class<?> pType, final String value) {
262     // By default use a phrase query is surrounded by "
263     String interpretedValue = APOSTROPHE + value + APOSTROPHE;
264     if (Enum.class.isAssignableFrom(pType)) {
265       // treat country codes special, they use iso codes
266       Enum<?> e;
267       if (Country.class.isAssignableFrom(pType)) {
268         e = Country.fromIsoCode(value);
269       } else {
270         e = VocabularyUtils.lookupEnum(value, (Class<? extends Enum<?>>) pType);
271       }
272       if (value == null) {
273         throw new IllegalArgumentException("Value Null is invalid for filter parameter " + pType.getName());
274       }
275       interpretedValue = String.valueOf(e.ordinal());
276 
277     } else if (UUID.class.isAssignableFrom(pType)) {
278       interpretedValue = UUID.fromString(value).toString();
279 
280     } else if (Double.class.isAssignableFrom(pType)) {
281       interpretedValue = String.valueOf(Double.parseDouble(value));
282 
283     } else if (Integer.class.isAssignableFrom(pType)) {
284       interpretedValue = String.valueOf(Integer.parseInt(value));
285 
286     } else if (Boolean.class.isAssignableFrom(pType)) {
287       interpretedValue = String.valueOf(Boolean.parseBoolean(value));
288     }
289 
290     return interpretedValue;
291   }

293   /**
294    * @param field the solr field
295    * @param param the parameter to use on a per field basis
296    * @return per field facet parameter, e.g. f.dataset_type.facet.sort
297    */
298   private String perFieldParamName(String field, String param) {
299     return "f." + field + "." + param;
300   }

302   /**
303    * Helper method that sets the highlighting parameters.
304    *
305    * @param searchRequest the searchRequest used to extract the parameters
306    * @param solrQuery this object is modified by adding the facets parameters
307    */
308   private void setHighLightParams(SearchRequest<P> searchRequest, SolrQuery solrQuery) {
309     solrQuery.setHighlight(searchRequest.isHighlight());
310     solrQuery.setHighlightSnippets(NUM_HL_SNIPPETS);
311     solrQuery.setHighlightFragsize(HL_FRAGMENT_SIZE);
312     if (searchRequest.isHighlight()) {
313       for (String hlField : queryBuilder.getHighlightedFields()) {
314         solrQuery.addHighlightField(hlField);
315       }
316     }
317   }

319   /**
320    * Sets the primary sort order query information.
321    *
322    * @param solrQuery to be modified.
323    */
324   private void setPrimarySortOrder(SolrQuery solrQuery) {
325     if (primarySortOrder != null) {
326       for (Map.Entry<String, SolrQuery.ORDER> so : primarySortOrder.entrySet()) {
327         solrQuery.addSort(so.getKey(), so.getValue());
328       }
329     }

332   /**
333    * Sets the query filters.
334    * Takes a {@link Multimap} and produces a Solr filter query containing the filtering criteria.
335    * The output string will be in the format: (param1:vp1.1 OR param1:vp1.2) AND (param2:vf2.1 OR param2:vp2.2)...
336    * The param-i key are taken from the key sets of the map and the vpi.j (value j of param i) are the entry set of the
337    * map.
338    *
339    * @return the String containing a Solr filter query
340    */
341   private void setQueryFilter(final SearchRequest<P> searchRequest, SolrQuery solrQuery) {
342     final Multimap<P, String> params = searchRequest.getParameters();
343     final boolean isFacetedRequest = FacetedSearchRequest.class.isAssignableFrom(searchRequest.getClass());
344     if (params != null) {
345       BiMap<P, String> fieldsPropertiesMapInv = facetFieldsPropertiesMap.inverse();
346       for (P param : params.keySet()) {
347         if (param != null && !fieldsPropertiesMapInv.containsKey(param)) {
348           LOG.warn("Unknown search parameter {}", param);
349           continue;
350         }
351         // solr field for this parameter
352         String solrFieldName = fieldsPropertiesMapInv.get(param);
353 
354         List<String> filterQueriesComponents = Lists.newArrayList();
355         for (String paramValue : params.get(param)) {
356           // Negate expression is removed if it is present
357           String interpretedValue = getInterpretedValue(param.type(), removeNegation(paramValue));
358           String queryComponent = PARAMS_JOINER.join(solrFieldName, interpretedValue);
359 
360           filterQueriesComponents.add(isNegated(paramValue) ? NOT_OP + queryComponent : queryComponent);
361         }
362         if (!filterQueriesComponents.isEmpty()) { // there are filter queries for this parameter
363           StringBuilder filterQuery = buildFilterQuery(isFacetedRequest, solrFieldName, filterQueriesComponents);
364           solrQuery.addFilterQuery(filterQuery.toString());
365         }
366 
367       }
368     }
