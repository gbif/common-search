3   import java.util.List;

5   /**
6    * A simple facade allowing a property to be a list of complex objects that can be treated as a simple String list
7    * for search highlighting. Return this interface in the getter of the annotated result object for the response builder
8    * to detect it.
9    */
10  public interface HighlightableList {
11  
12    /**
13     * Returns a list of string values in the same order as the original wrapped list the individual values are
14     * based on.
15     * @return a list of string values
16     */
17    List<String> getValues();
18  
19    /**
20     * Replaces an existing value in the value list with a new value, e.g. containing markup for highlighting.
21     * @param index the position in the list of values
22     * @param newValue the new value to set
23     */
24    void replaceValue(int index, String newValue);
25  }
