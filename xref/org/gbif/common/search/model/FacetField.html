13  package org.gbif.common.search.model;
14  
15  import java.lang.annotation.ElementType;
16  import java.lang.annotation.Retention;
17  import java.lang.annotation.RetentionPolicy;
18  import java.lang.annotation.Target;
19  
20  /**
21   * Annotation that defines a faceted field.
22   * The field corresponds to the name of it in the index data store or in the model object.
23   * The name is the actual name of the facet, this field is used to naming the facet in consistent way without
24   * dependencies of the field name. The intended use of name is that it should contain values coming from literal names
25   * of a enumerated type (e.g: Enum.name()).
26   */
27  @Target({ElementType.METHOD, ElementType.TYPE})
28  @Retention(RetentionPolicy.RUNTIME)
29  public @interface FacetField {
30  
31    /**
32     * Valid sort orders.
33     */
34    public enum SortOrder {
35      /**
36       * Sort the constraints by count (highest count first).
37       */
38      COUNT,
39      /**
40       * Sorted in index order (lexicographic by indexed term).
41       * For terms in the ascii range, this will be alphabetically sorted.
42       */
43      INDEX
44    }
45  
46    /**
47     * Indicates what type of algorithm/method to use when faceting a field.
48     */
49    public enum Method {
50  
51      /**
52       * Enumerates all terms in a field, calculating the set intersection of documents that match the term with documents
53       * that match the query.
54       */
55      ENUM,
56  
57      /**
58       * The facet counts are calculated by iterating over documents that match the query and summing the terms that
59       * appear in each document.
60       */
61      FIELD_CACHE,
62  
63      /**
64       * Works the same as FIELD_CACHE except the underlying cache data structure is built for each segment of the index
65       * individually.
66       */
67      FIELD_CACHE_SEGMENT
68    }
69  
70    /**
71     * @return the field name
72     */
73    String field();
74  
75    /**
76     * Indicates if count of all matching results which have no value for the field should be included.
77     * 
78     * @return flag that indicates missing facet is included
79     */
80    boolean missing() default false;
81  
82    /**
83     * @return the name of the facet exactly as SearchParameter enum specifies
84     */
85    String name();
86  
87    /**
88     * @return the sort order for this facet field
89     */
90    SortOrder sort() default SortOrder.COUNT;
91  
92    Method method() default Method.FIELD_CACHE;
93  }
