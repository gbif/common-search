21  /**
22   * Guice module providing a remote or embedded Solr server based on the modules configuration.
23   * Expects 5 properties to be bound:
24   * solr.server.type: must be set using literals available in {@link SolrServerType}, being SolrServerType.HTTP the
25   * default.
26   * solr.server: a http url to a remote Solr server, path for an embedded Solr or Zookeeper server url.
27   * solr.collection: Solr collection name (required by {@link CloudSolrServerBuilder})
28   * solr.server.http_lbservers: list of Solr http servers to create instance of {@link LBHttpSolrClient} (optional for
29   * {@link CloudSolrServerBuilder}).
30   * If empty an embedded server in /tmp will be created
31   * solr.delete: if null or true deletes any embedded server artifacts after java shuts down.
32   */
33  public class SolrModule extends PrivateModule {
34  
35    private static final Logger LOG = LoggerFactory.getLogger(SolrModule.class);
36  
37    /**
38     * Provides a Solr client instance based on the SolrServerType.
39     */
40    @Provides
41    @Singleton
42    @Inject
43    public SolrClient providerSolr(SolrConfig cfg) {
44      if (cfg.serverType == SolrServerType.EMBEDDED) {
45        LOG.info("Using embedded solr server {}", cfg.serverHome);
46        EmbeddedServerBuilder serverBuilder = new EmbeddedServerBuilder()
47            .withServerHomeDir(cfg.serverHome).withDeleteOnExit(cfg.deleteOnExit).withCoreName(cfg.collection);
48        return serverBuilder.build();
49      } else if (cfg.serverType == SolrServerType.LBHTTP) {
50        try {
51          LOG.info("Using remote load-balanced solr server {}", cfg.serverHome);
52          return new LBHttpSolrClient(cfg.serverHome);
53        } catch (MalformedURLException e) {
54          throw new IllegalStateException(e);
55        }
56      } else if (cfg.serverType == SolrServerType.CLOUD) {
57        CloudSolrServerBuilder serverBuilder = new CloudSolrServerBuilder()
58            .withZkHost(cfg.serverHome).withDefaultCollection(cfg.collection);
59        return serverBuilder.build();
60      } else { // cfg.serverType == SolrServerType.HTTP)
61        LOG.info("Using remote solr server {}", cfg.serverHome);
62        return new HttpSolrClient(cfg.serverHome);
63      }
64    }
65  
66    @Override
67    protected void configure() {
68      bind(SolrConfig.class).in(Scopes.SINGLETON);
69      expose(SolrClient.class);
70    }
71  }
